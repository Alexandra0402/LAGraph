14 // This is an Advanced algorithm. G->AT and G->out_degree are required for
15 // this method to use push-pull optimization. If not provided, this method
16 // defaults to a push-only algorithm, which can be slower. This is not
17 // user-callable (see LAGr_BreadthFirstSearch instead). G->AT and
18 // G->out_degree are not computed if not present.

47 int LG_BreadthFirstSearch_SSGrB
48 (
49  GrB_Vector *level,
50  GrB_Vector *parent,
51  const LAGraph_Graph G,
52  GrB_Index src,
53  char *msg
54 )
55 {
56 
57  //--------------------------------------------------------------------------
58  // check inputs
59  //--------------------------------------------------------------------------
60 
61 #if !LAGRAPH_SUITESPARSE
62  LG_ASSERT (false, GrB_NOT_IMPLEMENTED) ;
63 #else
64 
65  LG_CLEAR_MSG ;
66  GrB_Vector q = NULL ; // the current frontier
67  GrB_Vector w = NULL ; // to compute work remaining
68  GrB_Vector pi = NULL ; // parent vector
69  GrB_Vector v = NULL ; // level vector
70 
71  bool compute_level = (level != NULL) ;
72  bool compute_parent = (parent != NULL) ;
73  if (compute_level ) (*level ) = NULL ;
74  if (compute_parent) (*parent) = NULL ;
75  LG_ASSERT_MSG (compute_level || compute_parent, GrB_NULL_POINTER,
76  "either level or parent must be non-NULL") ;
77 
78  LG_TRY (LAGraph_CheckGraph (G, msg)) ;
79 
80  //--------------------------------------------------------------------------
81  // get the problem size and cached properties
82  //--------------------------------------------------------------------------
83 
84  GrB_Matrix A = G->A ;
85 
86  GrB_Index n, nvals ;
87  GRB_TRY (GrB_Matrix_nrows (&n, A)) ;
88  LG_ASSERT_MSG (src < n, GrB_INVALID_INDEX, "invalid source node") ;
89 
90  GRB_TRY (GrB_Matrix_nvals (&nvals, A)) ;
91 
92  GrB_Matrix AT = NULL ;
93  GrB_Vector Degree = G->out_degree ;
94  if (G->kind == LAGraph_ADJACENCY_UNDIRECTED ||
95  (G->kind == LAGraph_ADJACENCY_DIRECTED &&
96  G->is_symmetric_structure == LAGraph_TRUE))
97  {
98  // AT and A have the same structure and can be used in both directions
99  AT = G->A ;
100  }
101  else
102  {
103  // AT = A' is different from A. If G->AT is NULL, then a push-only
104  // method is used.
105  AT = G->AT ;
106  }
107 
108  // direction-optimization requires G->AT (if G is directed) and
109  // G->out_degree (for both undirected and directed cases)
110  bool push_pull = (Degree != NULL && AT != NULL) ;
111 
112  // determine the semiring type
113  GrB_Type int_type = (n > INT32_MAX) ? GrB_INT64 : GrB_INT32 ;
114  GrB_Semiring semiring ;
115 
116  if (compute_parent)
117  {
118  // use the ANY_SECONDI_INT* semiring: either 32 or 64-bit depending on
119  // the # of nodes in the graph.
120  semiring = (n > INT32_MAX) ?
121  GxB_ANY_SECONDI_INT64 : GxB_ANY_SECONDI_INT32 ;
122 
123  // create the parent vector. pi(i) is the parent id of node i
124  GRB_TRY (GrB_Vector_new (&pi, int_type, n)) ;
125  GRB_TRY (GxB_set (pi, GxB_SPARSITY_CONTROL, GxB_BITMAP + GxB_FULL)) ;
126  // pi (src) = src denotes the root of the BFS tree
127  GRB_TRY (GrB_Vector_setElement (pi, src, src)) ;
128 
129  // create a sparse integer vector q, and set q(src) = src
130  GRB_TRY (GrB_Vector_new (&q, int_type, n)) ;
131  GRB_TRY (GrB_Vector_setElement (q, src, src)) ;
132  }
133  else
134  {
135  // only the level is needed, use the LAGraph_any_one_bool semiring
136  semiring = LAGraph_any_one_bool ;
137 
138  // create a sparse boolean vector q, and set q(src) = true
139  GRB_TRY (GrB_Vector_new (&q, GrB_BOOL, n)) ;
140  GRB_TRY (GrB_Vector_setElement (q, true, src)) ;
141  }
142 
143  if (compute_level)
144  {
145  // create the level vector. v(i) is the level of node i
146  // v (src) = 0 denotes the source node
147  GRB_TRY (GrB_Vector_new (&v, int_type, n)) ;
148  GRB_TRY (GxB_set (v, GxB_SPARSITY_CONTROL, GxB_BITMAP + GxB_FULL)) ;
149  GRB_TRY (GrB_Vector_setElement (v, 0, src)) ;
150  }
151 
152  // workspace for computing work remaining
153  GRB_TRY (GrB_Vector_new (&w, GrB_INT64, n)) ;
154 
155  GrB_Index nq = 1 ; // number of nodes in the current level
156  double alpha = 8.0 ;
157  double beta1 = 8.0 ;
158  double beta2 = 512.0 ;
159  int64_t n_over_beta1 = (int64_t) (((double) n) / beta1) ;
160  int64_t n_over_beta2 = (int64_t) (((double) n) / beta2) ;
161 
162  //--------------------------------------------------------------------------
163  // BFS traversal and label the nodes
164  //--------------------------------------------------------------------------
165 
166  bool do_push = true ; // start with push
167  GrB_Index last_nq = 0 ;
168  int64_t edges_unexplored = nvals ;
169  bool any_pull = false ; // true if any pull phase has been done
170 
171  // {!mask} is the set of unvisited nodes
172  GrB_Vector mask = (compute_parent) ? pi : v ;
173 
174  for (int64_t nvisited = 1, k = 1 ; nvisited < n ; nvisited += nq, k++)
175  {
176 
177  //----------------------------------------------------------------------
178  // select push vs pull
179  //----------------------------------------------------------------------
180 
181  if (push_pull)
182  {
183  if (do_push)
184  {
185  // check for switch from push to pull
186  bool growing = nq > last_nq ;
187  bool switch_to_pull = false ;
188  if (edges_unexplored < n)
189  {
190  // very little of the graph is left; disable the pull
191  push_pull = false ;
192  }
193  else if (any_pull)
194  {
195  // once any pull phase has been done, the # of edges in the
196  // frontier has no longer been tracked. But now the BFS
197  // has switched back to push, and we're checking for yet
198  // another switch to pull. This switch is unlikely, so
199  // just keep track of the size of the frontier, and switch
200  // if it starts growing again and is getting big.
201  switch_to_pull = (growing && nq > n_over_beta1) ;
202  }
203  else
204  {
205  // update the # of unexplored edges
206  // w<q>=Degree
207  // w(i) = outdegree of node i if node i is in the queue
208  GRB_TRY (GrB_assign (w, q, NULL, Degree, GrB_ALL, n,
209  GrB_DESC_RS)) ;
210  // edges_in_frontier = sum (w) = # of edges incident on all
211  // nodes in the current frontier
212  int64_t edges_in_frontier = 0 ;
213  GRB_TRY (GrB_reduce (&edges_in_frontier, NULL,
214  GrB_PLUS_MONOID_INT64, w, NULL)) ;
215  edges_unexplored -= edges_in_frontier ;
216  switch_to_pull = growing &&
217  (edges_in_frontier > (edges_unexplored / alpha)) ;
218  }
219  if (switch_to_pull)
220  {
221  // switch from push to pull
222  do_push = false ;
223  }
224  }
225  else
226  {
227  // check for switch from pull to push
228  bool shrinking = nq < last_nq ;
229  if (shrinking && (nq <= n_over_beta2))
230  {
231  // switch from pull to push
232  do_push = true ;
233  }
234  }
235  any_pull = any_pull || (!do_push) ;
236  }
237 
238  //----------------------------------------------------------------------
239  // q = kth level of the BFS
240  //----------------------------------------------------------------------
241 
242  int sparsity = do_push ? GxB_SPARSE : GxB_BITMAP ;
243  GRB_TRY (GxB_set (q, GxB_SPARSITY_CONTROL, sparsity)) ;
244 
245  // mask is pi if computing parent, v if computing just level
246  if (do_push)
247  {
248  // push (saxpy-based vxm): q'{!mask} = q'*A
249  GRB_TRY (GrB_vxm (q, mask, NULL, semiring, q, A, GrB_DESC_RSC)) ;
250  }
251  else
252  {
253  // pull (dot-product-based mxv): q{!mask} = AT*q
254  GRB_TRY (GrB_mxv (q, mask, NULL, semiring, AT, q, GrB_DESC_RSC)) ;
255  }
256 
257  //----------------------------------------------------------------------
258  // done if q is empty
259  //----------------------------------------------------------------------
260 
261  last_nq = nq ;
262  GRB_TRY (GrB_Vector_nvals (&nq, q)) ;
263  if (nq == 0)
264  {
265  break ;
266  }
267 
268  //----------------------------------------------------------------------
269  // assign parents/levels
270  //----------------------------------------------------------------------
271 
272  if (compute_parent)
273  {
274  // q(i) currently contains the parent id of node i in tree.
275  // pi{q} = q
276  GRB_TRY (GrB_assign (pi, q, NULL, q, GrB_ALL, n, GrB_DESC_S)) ;
277  }
278  if (compute_level)
279  {
280  // v{q} = k, the kth level of the BFS
281  GRB_TRY (GrB_assign (v, q, NULL, k, GrB_ALL, n, GrB_DESC_S)) ;
282  }
283  }
284 
285  //--------------------------------------------------------------------------
286  // free workspace and return result
287  //--------------------------------------------------------------------------
288 
289  if (compute_parent) (*parent) = pi ;
290  if (compute_level ) (*level ) = v ;
291  LG_FREE_WORK ;
292  return (GrB_SUCCESS) ;
293 #endif
294 }
