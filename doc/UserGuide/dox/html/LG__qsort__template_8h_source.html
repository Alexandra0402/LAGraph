24 // LG_partition: use a pivot to partition an array
25 //------------------------------------------------------------------------------
26 
27 // C.A.R Hoare partition method, partitions an array in-place via a pivot.
28 // k = partition (A, n) partitions A [0:n-1] such that all entries in
29 // A [0:k] are <= all entries in A [k+1:n-1].
30 
31 static inline int64_t LG_partition
32 (
33  LG_args (A), // array(s) to partition
34  const int64_t n, // size of the array(s) to partition
35  uint64_t *seed, // random number seed, modified on output
36  LG_void *tx
37 )
38 {
39 
40  // select a pivot at random
41  int64_t pivot = ((n < LG_RANDOM15_MAX) ?
42  LG_Random15 (seed) : LG_Random60 (seed)) % n ;
43 
44  // get the Pivot
45  int64_t Pivot_0 [1] ; Pivot_0 [0] = A_0 [pivot] ;
46  #if LG_K > 1
47  int64_t Pivot_1 [1] ; Pivot_1 [0] = A_1 [pivot] ;
48  #endif
49  #if LG_K > 2
50  int64_t Pivot_2 [1] ; Pivot_2 [0] = A_2 [pivot] ;
51  #endif
52 
53  // At the top of the while loop, A [left+1...right-1] is considered, and
54  // entries outside this range are in their proper place and not touched.
55  // Since the input specification of this function is to partition A
56  // [0..n-1], left must start at -1 and right must start at n.
57  int64_t left = -1 ;
58  int64_t right = n ;
59 
60  // keep partitioning until the left and right sides meet
61  while (true)
62  {
63  // loop invariant: A [0..left] < pivot and A [right..n-1] > pivot,
64  // so the region to be considered is A [left+1 ... right-1].
65 
66  // increment left until finding an entry A [left] >= pivot
67  do { left++ ; } while (LG_lt (A, left, Pivot, 0)) ;
68 
69  // decrement right until finding an entry A [right] <= pivot
70  do { right-- ; } while (LG_lt (Pivot, 0, A, right)) ;
71 
72  // now A [0..left-1] < pivot and A [right+1..n-1] > pivot, but
73  // A [left] > pivot and A [right] < pivot, so these two entries
74  // are out of place and must be swapped.
75 
76  // However, if the two sides have met, the partition is finished.
77  if (left >= right)
78  {
79  // A has been partitioned into A [0:right] and A [right+1:n-1].
80  // k = right+1, so A is split into A [0:k-1] and A [k:n-1].
81  return (right + 1) ;
82  }
83 
84  // since A [left] > pivot and A [right] < pivot, swap them
85  LG_swap (A, left, right) ;
86 
87  // after the swap this condition holds:
88  // A [0..left] < pivot and A [right..n-1] > pivot
89  }
90 }

93 // LG_quicksort: recursive single-threaded quicksort
94 //------------------------------------------------------------------------------

96 static void LG_quicksort // sort A [0:n-1]
97 (
98  LG_args (A), // array(s) to sort
99  const int64_t n, // size of the array(s) to sort
100  uint64_t *seed, // random number seed
101  LG_void *tx
102 )
103 {
104 
105  if (n < 20)
106  {
107  // in-place insertion sort on A [0:n-1], where n is small
108  for (int64_t k = 1 ; k < n ; k++)
109  {
110  for (int64_t j = k ; j > 0 && LG_lt (A, j, A, j-1) ; j--)
111  {
112  // swap A [j-1] and A [j]
113  LG_swap (A, j-1, j) ;
114  }
115  }
116  }
117  else
118  {
119  // partition A [0:n-1] into A [0:k-1] and A [k:n-1]
120  int64_t k = LG_partition (LG_arg (A), n, seed, tx) ;
121 
122  // sort each partition
123  LG_quicksort (LG_arg (A), k, seed, tx) ; // sort A [0:k-1]
124  LG_quicksort (LG_arg_offset (A, k), n-k, seed, tx) ; // sort A [k+1:n-1]
125  }
126 }
