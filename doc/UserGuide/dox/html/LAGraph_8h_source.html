12 // LAGraph is a package of graph algorithms based on GraphBLAS. GraphBLAS
13 // defines a set of sparse matrix operations on an extended algebra of
14 // semirings, using an almost unlimited variety of operators and types. When
15 // applied to sparse adjacency matrices, these algebraic operations are
16 // equivalent to computations on graphs. GraphBLAS provides a powerful and
17 // expressive framework creating graph algorithms based on the elegant
18 // mathematics of sparse matrix operations on a semiring.
19 
20 // However, GraphBLAS itself does not have graph algorithms. The purpose of
21 // LAGraph is to provide a robust, easy-to-use high-performance library of
22 // graph algorithms that rely on GraphBLAS.
23

114 // All LAGraph functions that return an int use it to indicate an error status
115 // (described below), where zero (GrB_SUCCESS) denotes success, negative values
116 // indicate an error, and positive values denote success but with some kind of
117 // algorithm-specific note or warning.
118 
119 // In addition, all such LAGraph functions also have a final msg parameter that
120 // is a pointer to a user-allocated string in which an algorithm-specific error
121 // message can be returned. If msg is NULL, no error message is returned.
122 // This is not itself an error condition, it just indicates that the caller
123 // does not need the message returned. If the message string is provided but
124 // no error occurs, an empty string is returned.
125 
126 // A single LAGraph function has no error handling mechanism
127 // (LAGraph_WallClockTime); it returns a value of type double, and does not
128 // have an final msg string parameter.
129 
130 // LAGRAPH_MSG_LEN: The maximum required length of a message string
131 #define LAGRAPH_MSG_LEN 256
132 
133 // For example, the following call computes the breadth-first-search of an
134 // LAGraph_Graph G, starting at a given source node. It returns a status of
135 // zero if it succeeds and a negative value on failure.
136 
137 /*
138  GrB_Vector level, parent ;
139  char msg [LAGRAPH_MSG_LEN] ;
140  int status = LAGr_BreadthFirstSearch (&level, &parent, G, src, msg) ;
141  if (status < 0)
142  {
143  printf ("error: %s\n", msg) ;
144  // take corrective action ...
145  }
146 */

148 //------------------------------------------------------------------------------
149 // LAGraph error status:
150 //------------------------------------------------------------------------------
151 
152 // All LAGraph methods return an int to denote their status: zero if they are
153 // successful (which is the value of GrB_SUCCESS), negative on error, or
154 // positive for an informational value (such as GrB_NO_VALUE). Integers in the
155 // range -999 to 999 are reserved for GraphBLAS GrB_Info return values:
156 
157 // successful results:
158 // GrB_SUCCESS = 0 // all is well
159 // GrB_NO_VALUE = 1 // A(i,j) requested but not there
160 
161 // errors:
162 // GrB_UNINITIALIZED_OBJECT = -1 // object has not been initialized
163 // GrB_NULL_POINTER = -2 // input pointer is NULL
164 // GrB_INVALID_VALUE = -3 // generic error; some value is bad
165 // GrB_INVALID_INDEX = -4 // row or column index is out of bounds
166 // GrB_DOMAIN_MISMATCH = -5 // object domains are not compatible
167 // GrB_DIMENSION_MISMATCH = -6 // matrix dimensions do not match
168 // GrB_OUTPUT_NOT_EMPTY = -7 // output matrix already has values
169 // GrB_NOT_IMPLEMENTED = -8 // method not implemented
170 // GrB_PANIC = -101 // unknown error
171 // GrB_OUT_OF_MEMORY = -102 // out of memory
172 // GrB_INSUFFICIENT_SPACE = -103, // output array not large enough
173 // GrB_INVALID_OBJECT = -104 // object is corrupted
174 // GrB_INDEX_OUT_OF_BOUNDS = -105 // row or col index out of bounds
175 // GrB_EMPTY_OBJECT = -106 // an object does not contain a value
176 
177 // LAGraph returns any errors it receives from GraphBLAS, and also uses the
178 // GrB_* error codes in these cases:
179 // GrB_INVALID_INDEX: if a source node id is out of range
180 // GrB_INVALID_VALUE: if an enum to select a method or option is out of range
181 // GrB_NOT_IMPLEMENTED: if a type is not supported, or when SuiteSparse
182 // GraphBLAS is required.
183 
184 // Many LAGraph methods share common error cases, described below. These
185 // return values are in the range -1000 to -1999. Return values of -2000 or
186 // greater may be used by specific LAGraph methods, which denote errors not in
187 // this list:
188 
189 // LAGRAPH_INVALID_GRAPH: the input graph is invalid; the details are given
190 // in the error msg string returned by the method.
191 
192 // LAGRAPH_SYMMETRIC_STRUCTURE_REQUIRED: the method requires an undirected
193 // graph, or a directed graph with an adjacency matrix that is known to
194 // have a symmetric structure. LAGraph_Cached_IsSymmetricStructure can
195 // be used to determine this cached property.
196 
197 // LAGRAPH_IO_ERROR: a file input or output method failed, or an input file
198 // has an incorrect format that cannot be parsed.
199 
200 // LAGRAPH_NOT_CACHED: some methods require one or more cached properties to
201 // be computed before calling them (AT, out_degree, or in_degree. Use
202 // LAGraph_Cached_AT, LAGraph_Cached_OutDegree, and/or
203 // LAGraph_Cached_InDegree to compute them.
204 
205 // LAGRAPH_NO_SELF_EDGES_ALLOWED: some methods requires that the graph have
206 // no self edges, which correspond to the entries on the diagonal of the
207 // adjacency matrix. If the G->nself_edges cached property is nonzero or
208 // unknown, this error condition is returned. Use
209 // LAGraph_Cached_NSelfEdges to compute G->nself_edges, or
210 // LAGraph_DeleteSelfEdges to remove all diagonal entries from G->A.
211 
212 // LAGRAPH_CONVERGENCE_FAILURE: an iterative process failed to converge to
213 // a good solution.
214 
215 // LAGRAPH_CACHE_NOT_NEEDED: this is a warning, not an error. It is returned
216 // by LAGraph_Cached_* methods when asked to compute cached properties
217 // that are not needed. These include G->AT and G->in_degree for an
218 // undirected graph.
219 
220 // LAGraph errors:
221 #define LAGRAPH_INVALID_GRAPH (-1000)
222 #define LAGRAPH_SYMMETRIC_STRUCTURE_REQUIRED (-1001)
223 #define LAGRAPH_IO_ERROR (-1002)
224 #define LAGRAPH_NOT_CACHED (-1003)
225 #define LAGRAPH_NO_SELF_EDGES_ALLOWED (-1004)
226 #define LAGRAPH_CONVERGENCE_FAILURE (-1005)
227 
228 // LAGraph warnings:
229 #define LAGRAPH_CACHE_NOT_NEEDED ( 1000)

231 // Summary of return values for all LAGraph functions that return int:
232 // * GrB_SUCCESS if successful
233 // * a negative GrB_Info value on error (in range -999 to -1)
234 // * a positive GrB_Info value if successful but with extra information
235 // (in range 1 to 999)
236 // * -1999 to -1000: a common LAGraph-specific error, see list above
237 // * 1000 to 1999: if successful, with extra LAGraph-specific information
238 // * <= -2000: an LAGraph error specific to a particular LAGraph method
239 // * >= 2000: an LAGraph warning specific to a particular LAGraph method

241 //------------------------------------------------------------------------------
242 // LAGRAPH_TRY: try an LAGraph method and check for errors
243 //------------------------------------------------------------------------------
244 
245 // In a robust application, the return values from each call to LAGraph and
246 // GraphBLAS should be checked, and corrective action should be taken if an
247 // error occurs. The LAGRAPH_TRY and GRB_TRY macros assist in this effort.
248 
249 // LAGraph and GraphBLAS are written in C, and so they cannot rely on the
250 // try/catch mechanism of C++. To accomplish a similar goal, each LAGraph file
251 // must #define its own file-specific macro called LAGRAPH_CATCH. The typical
252 // usage of macro is to free any temporary matrices/vectors or workspace when
253 // an error occurs, and then "throw" the error by returning to the caller. A
254 // user application may also #define LAGRAPH_CATCH and use these macros.
255 
256 // A typical example of a user function that calls LAGraph might #define
257 // LAGRAPH_CATCH as follows. Suppose workvector is a GrB_vector used for
258 // computations internal to the mybfs function, and W is a (double *) space
259 // created by malloc.
260 
261 #if example_usage_only
262 
263  // an example user-defined LAGRAPH_CATCH macro
264  #define LAGRAPH_CATCH(status) \
265  { \
266  /* an LAGraph error has occurred */ \
267  printf ("LAGraph error: (%d): file: %s, line: %d\n%s\n", \
268  status, __FILE__, __LINE__, msg) ; \
269  /* free any internal workspace and return the status */ \
270  GrB_free (*parent) ; \
271  GrB_free (workvector) ; \
272  LAGraph_Free ((void **) &W, NULL) ; \
273  return (status) ; \
274  }
275 
276  // an example user function that uses LAGRAPH_TRY / LAGRAPH_CATCH
277  int mybfs (LAGraph_Graph G, GrB_Vector *parent, int64_t src)
278  {
279  GrB_Vector workvector = NULL ;
280  double *W = NULL ;
281  char msg [LAGRAPH_MSG_LEN] ;
282  (*parent) = NULL ;
283  LAGRAPH_TRY (LAGr_BreadthFirstSearch (NULL, parent, G, src, true,
284  msg)) ;
285  // ...
286  return (GrB_SUCCESS) ;
287  }
288 
289 #endif
290 
291 #define LAGRAPH_TRY(LAGraph_method) \
292 { \
293  int LG_status = LAGraph_method ; \
294  if (LG_status < GrB_SUCCESS) \
295  { \
296  LAGRAPH_CATCH (LG_status) ; \
297  } \
298 }
299 
300 //------------------------------------------------------------------------------
301 // GRB_TRY: try a GraphBLAS method and check for errors
302 //------------------------------------------------------------------------------
303 
304 // LAGraph provides a similar functionality for calling GraphBLAS methods.
305 // GraphBLAS returns info = 0 (GrB_SUCCESS) or 1 (GrB_NO_VALUE) on success, and
306 // a value < 0 on failure. The user application must #define GRB_CATCH to use
307 // GRB_TRY. Note that GraphBLAS_info is internal to this macro. If the
308 // user application or LAGraph method wants a copy, a statement such as
309 // info = GraphBLAS_info ; where info is defined outside of this macro.
310 
311 // GraphBLAS and LAGraph both use the convention that negative values are
312 // errors, and the LAGraph_status is a superset of the GrB_Info enum. As a
313 // result, the user can define LAGRAPH_CATCH and GRB_TRY as the same operation.
314 // The main difference between the two would be the error message string. For
315 // LAGraph, the string is the last parameter, and LAGRAPH_CATCH can optionally
316 // print it out. For GraphBLAS, the GrB_error mechanism can return a string.
317 
318 #define GRB_TRY(GrB_method) \
319 { \
320  GrB_Info LG_GrB_Info = GrB_method ; \
321  if (LG_GrB_Info < GrB_SUCCESS) \
322  { \
323  GRB_CATCH (LG_GrB_Info) ; \
324  } \
325 }

556 // LAGraph_Graph: the primary graph data structure of LAGraph
557 //------------------------------------------------------------------------------
558 
569 /* (1) primary components:
570  * A the adjacency matrix of the graph
571  * kind the kind of graph (undirected, directed, bipartite, ...)
572  * (2) cached properties:
573  * AT AT = A'
574  * out_degree out_degree(i) = # of entries in A(i,:)
575  * in_degree in_degree(j) = # of entries in A(:,j)
576  * is_symmetric_structure: true if the structure of A is symmetric
577  * nself_edges the number of entries on the diagonal of A
578  * emin minimum edge weight
579  * emax maximum edge weight
580  */
581 
582 struct LAGraph_Graph_struct
583 {
584 
585  //--------------------------------------------------------------------------
586  // primary components of the graph
587  //--------------------------------------------------------------------------
588 
591 
592  GrB_Matrix A ;
593  LAGraph_Kind kind ;
594 
596 
597  // possible future components:
598  // multigraph ..
599  // GrB_Matrix *Amult ; // array of size nmatrices
600  // int nmatrices ;
601  // GrB_Vector VertexWeights ;
602 
603  //--------------------------------------------------------------------------
604  // cached properties of the graph
605  //--------------------------------------------------------------------------
606 
607  // All of these components may be deleted or set to 'unknown' at any time.
608  // For example, if AT is NULL, then the transpose of A has not been
609  // computed. A scalar cached property of type LAGraph_Boolean would be set
610  // to LAGRAPH_UNKNOWN to denote that its value is unknown.
611 
612  // If present, the cached properties must be valid and accurate. If the
613  // graph changes, these cached properties can either be recomputed or
614  // deleted to denote the fact that they are unknown. This choice is up to
615  // individual LAGraph methods and utilities.
616 
617  // LAGraph methods can set non-scalar cached properties only if they are
618  // constructing the graph. They cannot modify them or create them if the
619  // graph is declared as a read-only object in the parameter list of the
620  // method.
621 
624 
625  GrB_Matrix AT ;
626 
627  GrB_Vector out_degree ;
628 
632  GrB_Vector in_degree ;
633 
639  // If G is held as an incidence matrix, then G->A might be rectangular,
640  // in the future, but the graph G may have a symmetric structure anyway.
641  LAGraph_Boolean is_symmetric_structure ;
642 
648  int64_t nself_edges ;
649 
653  GrB_Scalar emin ;
654  LAGraph_State emin_state ;
658 
659  GrB_Scalar emax ;
660  LAGraph_State emax_state ;
664 
666 
667  // possible future cached properties:
668 
669  // Some algorithms may want to know if the graph has any edge weights
670  // exactly equal to zero. In some cases, this can be inferred from the
671  // emin/emax bounds, or it can be indicated via the following cached
672  // property:
673  // LAGraph_Boolean nonzero ; // If true, then all entries in
674  // G->A are known to be nonzero. If false, G->A may contain
675  // entries in its structure that are identically equal to zero. If
676  // unknown, then G->A may or may not have entries equal to zero.
677  // other edge weight metrics: median, standard deviation.... Might be
678  // useful for computing Delta for a Basic SSSP.
679  // GrB_Vector row_sum, col_sum ;
680  // row_sum(i) = sum(abs(A(i,:))), regardless of kind
681  // col_sum(j) = sum(abs(A(:,j))), regardless of kind
682  // LAGraph_Boolean connected ; // true if G is a connected graph
683 } ;
