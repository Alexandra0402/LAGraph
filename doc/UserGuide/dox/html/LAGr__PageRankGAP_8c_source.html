14 // This is an Advanced algorithm (G->AT and G->out_degree are required).
15 
16 // PageRank for the GAP benchmark (only). Do not use in production.
17 
18 // This algorithm follows the specification given in the GAP Benchmark Suite:
19 // https://arxiv.org/abs/1508.03619 which assumes that both A and A' are
20 // already available, as are the row and column degrees. The GAP specification
21 // ignores dangling nodes (nodes with no outgoing edges, also called sinks),
22 // and thus shouldn't be used in production. This method is for the GAP
23 // benchmark only. See LAGr_PageRank for a method that
24 // handles sinks correctly. This method does not return a centrality metric
25 // such that sum(centrality) is 1, if sinks are present.
26 
27 // The G->AT and G->out_degree cached properties must be defined for this
28 // method. If G is undirected or G->A is known to have a symmetric structure,
29 // then G->A is used instead of G->AT, however.
30

37 }

47 int LAGr_PageRankGAP
48 (
49  // output:
50  GrB_Vector *centrality, // centrality(i): GAP-style pagerank of node i
51  int *iters, // number of iterations taken
52  // input:
53  const LAGraph_Graph G, // input graph
54  float damping, // damping factor (typically 0.85)
55  float tol, // stopping tolerance (typically 1e-4) ;
56  int itermax, // maximum number of iterations (typically 100)
57  char *msg
58 )
59 {
60 
61  //--------------------------------------------------------------------------
62  // check inputs
63  //--------------------------------------------------------------------------
64 
65  LG_CLEAR_MSG ;
66  GrB_Vector r = NULL, d = NULL, t = NULL, w = NULL, d1 = NULL ;
67  LG_ASSERT (centrality != NULL, GrB_NULL_POINTER) ;
68  LG_TRY (LAGraph_CheckGraph (G, msg)) ;
69  GrB_Matrix AT ;
70  if (G->kind == LAGraph_ADJACENCY_UNDIRECTED ||
71  G->is_symmetric_structure == LAGraph_TRUE)
72  {
73  // A and A' have the same structure
74  AT = G->A ;
75  }
76  else
77  {
78  // A and A' differ
79  AT = G->AT ;
80  LG_ASSERT_MSG (AT != NULL,
81  LAGRAPH_NOT_CACHED, "G->AT is required") ;
82  }
83  GrB_Vector d_out = G->out_degree ;
84  LG_ASSERT_MSG (d_out != NULL,
85  LAGRAPH_NOT_CACHED, "G->out_degree is required") ;
86 
87  //--------------------------------------------------------------------------
88  // initializations
89  //--------------------------------------------------------------------------
90 
91  GrB_Index n ;
92  (*centrality) = NULL ;
93  GRB_TRY (GrB_Matrix_nrows (&n, AT)) ;
94 
95  const float scaled_damping = (1 - damping) / n ;
96  const float teleport = scaled_damping ; // teleport = (1 - damping) / n
97  float rdiff = 1 ; // first iteration is always done
98 
99  // r = 1 / n
100  GRB_TRY (GrB_Vector_new (&t, GrB_FP32, n)) ;
101  GRB_TRY (GrB_Vector_new (&r, GrB_FP32, n)) ;
102  GRB_TRY (GrB_Vector_new (&w, GrB_FP32, n)) ;
103  GRB_TRY (GrB_assign (r, NULL, NULL, (float) (1.0 / n), GrB_ALL, n, NULL)) ;
104 
105  // prescale with damping factor, so it isn't done each iteration
106  // d = d_out / damping ;
107  GRB_TRY (GrB_Vector_new (&d, GrB_FP32, n)) ;
108  GRB_TRY (GrB_apply (d, NULL, NULL, GrB_DIV_FP32, d_out, damping, NULL)) ;
109 
110  // d1 = 1 / damping
111  float dmin = 1.0 / damping ;
112  GRB_TRY (GrB_Vector_new (&d1, GrB_FP32, n)) ;
113  GRB_TRY (GrB_assign (d1, NULL, NULL, dmin, GrB_ALL, n, NULL)) ;
114  // d = max (d1, d)
115  GRB_TRY (GrB_eWiseAdd (d, NULL, NULL, GrB_MAX_FP32, d1, d, NULL)) ;
116  GrB_free (&d1) ;
117 
118  //--------------------------------------------------------------------------
119  // pagerank iterations
120  //--------------------------------------------------------------------------
121 
122  for ((*iters) = 0 ; (*iters) < itermax && rdiff > tol ; (*iters)++)
123  {
124  // swap t and r ; now t is the old score
125  GrB_Vector temp = t ; t = r ; r = temp ;
126  // w = t ./ d
127  GRB_TRY (GrB_eWiseMult (w, NULL, NULL, GrB_DIV_FP32, t, d, NULL)) ;
128  // r = teleport
129  GRB_TRY (GrB_assign (r, NULL, NULL, teleport, GrB_ALL, n, NULL)) ;
130  // r += A'*w
131  GRB_TRY (GrB_mxv (r, NULL, GrB_PLUS_FP32, LAGraph_plus_second_fp32,
132  AT, w, NULL)) ;
133  // t -= r
134  GRB_TRY (GrB_assign (t, NULL, GrB_MINUS_FP32, r, GrB_ALL, n, NULL)) ;
135  // t = abs (t)
136  GRB_TRY (GrB_apply (t, NULL, NULL, GrB_ABS_FP32, t, NULL)) ;
137  // rdiff = sum (t)
138  GRB_TRY (GrB_reduce (&rdiff, NULL, GrB_PLUS_MONOID_FP32, t, NULL)) ;
139  }
140 
141  //--------------------------------------------------------------------------
142  // free workspace and return result
143  //--------------------------------------------------------------------------
144 
145  (*centrality) = r ;
146  LG_FREE_WORK ;
147  return (GrB_SUCCESS) ;
148 }
