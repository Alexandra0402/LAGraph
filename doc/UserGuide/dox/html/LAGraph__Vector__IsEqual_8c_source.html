14 // LAGraph_Vector_IsEqual: checks if two vectors are identically equal (same
15 // size, type, structure, size, and values).
16 
17 // See also LAGraph_Matrix_IsEqual.
18 
19 // If the two vectors are GrB_FP32, GrB_FP64, or related, and have NaNs, then
20 // this function will return false, since NaN == NaN is false. To check for
21 // NaN equality (like isequalwithequalnans in MATLAB), use
22 // LAGraph_Vector_IsEqualOp with a user-defined operator f(x,y) that returns
23 // true if x and y are both NaN.
24 
25 #define LG_FREE_WORK GrB_free (&C) ;
26 
27 #include "LG_internal.h"
28 
29 int LAGraph_Vector_IsEqual
30 (
31  // output:
32  bool *result, // true if A == B, false if A != B or error
33  // input:
34  const GrB_Vector A,
35  const GrB_Vector B,
36  char *msg
37 )
38 {
39 
40  //--------------------------------------------------------------------------
41  // check inputs
42  //--------------------------------------------------------------------------
43 
44  LG_CLEAR_MSG ;
45  GrB_Vector C = NULL ;
46  LG_ASSERT (result != NULL, GrB_NULL_POINTER) ;
47 
48  //--------------------------------------------------------------------------
49  // check for NULL and aliased vectors
50  //--------------------------------------------------------------------------
51 
52  if (A == NULL || B == NULL || A == B)
53  {
54  // two NULL vectors are identical, as are two aliased matrices
55  (*result) = (A == B) ;
56  return (GrB_SUCCESS) ;
57  }
58 
59  //--------------------------------------------------------------------------
60  // compare the type of A and B
61  //--------------------------------------------------------------------------
62 
63  char atype_name [LAGRAPH_MAX_NAME_LEN] ;
64  char btype_name [LAGRAPH_MAX_NAME_LEN] ;
65  LG_TRY (LAGraph_Vector_TypeName (atype_name, A, msg)) ;
66  LG_TRY (LAGraph_Vector_TypeName (btype_name, B, msg)) ;
67  if (!MATCHNAME (atype_name, btype_name))
68  {
69  // types differ
70  (*result) = false ;
71  return (GrB_SUCCESS) ;
72  }
73 
74  //--------------------------------------------------------------------------
75  // compare the size of A and B
76  //--------------------------------------------------------------------------
77 
78  GrB_Index nrows1, nrows2;
79  GRB_TRY (GrB_Vector_size (&nrows1, A)) ;
80  GRB_TRY (GrB_Vector_size (&nrows2, B)) ;
81  if (nrows1 != nrows2)
82  {
83  // # of rows differ
84  (*result) = false ;
85  return (GrB_SUCCESS) ;
86  }
87 
88  //--------------------------------------------------------------------------
89  // compare the # entries in A and B
90  //--------------------------------------------------------------------------
91 
92  GrB_Index nvals1, nvals2 ;
93  GRB_TRY (GrB_Vector_nvals (&nvals1, A)) ;
94  GRB_TRY (GrB_Vector_nvals (&nvals2, B)) ;
95  if (nvals1 != nvals2)
96  {
97  // # of entries differ
98  (*result) = false ;
99  return (GrB_SUCCESS) ;
100  }
101 
102  //--------------------------------------------------------------------------
103  // get the GrB_EQ_type operator
104  //--------------------------------------------------------------------------
105 
106  GrB_Type type ;
107  LG_TRY (LAGraph_TypeFromName (&type, atype_name, msg)) ;
108  GrB_BinaryOp op = NULL ;
109  // select the comparator operator
110  if (type == GrB_BOOL ) op = GrB_EQ_BOOL ;
111  else if (type == GrB_INT8 ) op = GrB_EQ_INT8 ;
112  else if (type == GrB_INT16 ) op = GrB_EQ_INT16 ;
113  else if (type == GrB_INT32 ) op = GrB_EQ_INT32 ;
114  else if (type == GrB_INT64 ) op = GrB_EQ_INT64 ;
115  else if (type == GrB_UINT8 ) op = GrB_EQ_UINT8 ;
116  else if (type == GrB_UINT16) op = GrB_EQ_UINT16 ;
117  else if (type == GrB_UINT32) op = GrB_EQ_UINT32 ;
118  else if (type == GrB_UINT64) op = GrB_EQ_UINT64 ;
119  else if (type == GrB_FP32 ) op = GrB_EQ_FP32 ;
120  else if (type == GrB_FP64 ) op = GrB_EQ_FP64 ;
121  #if 0
122  else if (type == GxB_FC32 ) op = GxB_EQ_FC32 ;
123  else if (type == GxB_FC64 ) op = GxB_EQ_FC64 ;
124  #endif
125 
126  LG_ASSERT_MSG (op != NULL, GrB_NOT_IMPLEMENTED, "type not supported") ;
127 
128  //--------------------------------------------------------------------------
129  // C = A .* B, where the structure of C is the intersection of A and B
130  //--------------------------------------------------------------------------
131 
132  GRB_TRY (GrB_Vector_new (&C, GrB_BOOL, nrows1)) ;
133  GRB_TRY (GrB_eWiseMult (C, NULL, NULL, op, A, B, NULL)) ;
134 
135  //--------------------------------------------------------------------------
136  // ensure C has the same number of entries as A and B
137  //--------------------------------------------------------------------------
138 
139  GrB_Index nvals ;
140  GRB_TRY (GrB_Vector_nvals (&nvals, C)) ;
141  if (nvals != nvals1)
142  {
143  // structure of A and B are different
144  LG_FREE_WORK ;
145  (*result) = false ;
146  return (GrB_SUCCESS) ;
147  }
148 
149  //--------------------------------------------------------------------------
150  // result = and (C)
151  //--------------------------------------------------------------------------
152 
153  GRB_TRY (GrB_reduce (result, NULL, GrB_LAND_MONOID_BOOL, C, NULL)) ;
154 
155  //--------------------------------------------------------------------------
156  // free workspace and return result
157  //--------------------------------------------------------------------------
158 
159  LG_FREE_WORK ;
160  return (GrB_SUCCESS) ;
161 }
