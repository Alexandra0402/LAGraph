21 int LAGraph_Cached_EMin
22 (
23  // input/output:
24  LAGraph_Graph G, // graph to determine G->emin
25  char *msg
26 )
27 {
28 
29  //--------------------------------------------------------------------------
30  // clear msg and check G
31  //--------------------------------------------------------------------------
32 
33  LG_CLEAR_MSG_AND_BASIC_ASSERT (G, msg) ;
34 
35  if (G->emin != NULL)
36  {
37  // G->emin already computed
38  return (GrB_SUCCESS) ;
39  }
40 
41  G->emin_state = LAGRAPH_UNKNOWN ;
42 
43  //--------------------------------------------------------------------------
44  // determine the type of G->A and the corresponding min monoid
45  //--------------------------------------------------------------------------
46 
47  char atype_name [LAGRAPH_MAX_NAME_LEN] ;
48  LG_TRY (LAGraph_Matrix_TypeName (atype_name, G->A, msg)) ;
49  GrB_Type atype ;
50  LG_TRY (LAGraph_TypeFromName (&atype, atype_name, msg)) ;
51  GrB_Monoid monoid ;
52  if (atype == GrB_BOOL ) monoid = GrB_LAND_MONOID_BOOL ;
53  else if (atype == GrB_INT8 ) monoid = GrB_MIN_MONOID_INT8 ;
54  else if (atype == GrB_INT16 ) monoid = GrB_MIN_MONOID_INT16 ;
55  else if (atype == GrB_INT32 ) monoid = GrB_MIN_MONOID_INT32 ;
56  else if (atype == GrB_INT64 ) monoid = GrB_MIN_MONOID_INT64 ;
57  else if (atype == GrB_UINT8 ) monoid = GrB_MIN_MONOID_UINT8 ;
58  else if (atype == GrB_UINT16) monoid = GrB_MIN_MONOID_UINT16 ;
59  else if (atype == GrB_UINT32) monoid = GrB_MIN_MONOID_UINT32 ;
60  else if (atype == GrB_UINT64) monoid = GrB_MIN_MONOID_UINT64 ;
61  else if (atype == GrB_FP32 ) monoid = GrB_MIN_MONOID_FP32 ;
62  else if (atype == GrB_FP64 ) monoid = GrB_MIN_MONOID_FP64 ;
63  else
64  {
65  LG_ASSERT_MSG (false, GrB_NOT_IMPLEMENTED, "type not supported") ;
66  }
67 
68  //--------------------------------------------------------------------------
69  // compute G->emin
70  //--------------------------------------------------------------------------
71 
72  GRB_TRY (GrB_Scalar_new (&(G->emin), atype)) ;
73  GRB_TRY (GrB_reduce (G->emin, NULL, monoid, G->A, NULL)) ;
74  G->emin_state = LAGraph_VALUE ;
75  return (GrB_SUCCESS) ;
76 }
