15 // This is an Advanced algorithm (G->is_symmetric_structure must be known),
16 // but it is not user-callable (see LAGr_ConnectedComponents instead).
17 
18 // Code is based on Boruvka's minimum spanning forest algorithm.
19 
20 // This method relies solely on GrB* methods in the V2.0 C API, but it much
21 // slower in general than LG_CC_FastSV6, which uses GxB pack/unpack methods
22 // for faster access to the contents of the matrices and vectors.

30 // w[Px[i]] = min(w[Px[i]], s[i]) for i in [0..n-1].
31 
32 static GrB_Info Reduce_assign
33 (
34  GrB_Vector w, // input/output vector of size n
35  GrB_Vector s, // input vector of size n
36  GrB_Index *Px, // Px: array of size n
37  GrB_Index *mem, // workspace of size 3*n
38  GrB_Index n
39 )
40 {
41  char *msg = NULL ;
42  GrB_Index *ind = mem ;
43  GrB_Index *sval = ind + n ;
44  GrB_Index *wval = sval + n ;
45  GRB_TRY (GrB_Vector_extractTuples (ind, wval, &n, w)) ;
46  GRB_TRY (GrB_Vector_extractTuples (ind, sval, &n, s)) ;
47  for (GrB_Index j = 0 ; j < n ; j++)
48  {
49  if (sval [j] < wval [Px [j]])
50  {
51  wval [Px [j]] = sval [j] ;
52  }
53  }
54  GRB_TRY (GrB_Vector_clear (w)) ;
55  GRB_TRY (GrB_Vector_build (w, ind, wval, n, GrB_PLUS_UINT64)) ;
56  LG_FREE_ALL ;
57  return (GrB_SUCCESS) ;
58 }

60 //------------------------------------------------------------------------------
61 // select_func: IndexUnaryOp for pruning entries from S
62 //------------------------------------------------------------------------------
63 
64 // Rather than using a global variable to access the Px array, it is passed to
65 // the select function as a uint64_t value that contains a pointer to Px. It
66 // might be better to create a user-defined type for y, but this works fine.
67 
68 void my_select_func (void *z, const void *x,
69  const GrB_Index i, const GrB_Index j, const void *y)
70 {
71  GrB_Index *Px = (*(GrB_Index **) y) ;
72  (*((bool *) z)) = (Px [i] != Px [j]) ;
73 }

100 int LG_CC_Boruvka
101 (
102  // output:
103  GrB_Vector *component, // output: array of component identifiers
104  // input:
105  const LAGraph_Graph G, // input graph, not modified
106  char *msg
107 )
108 {
109 
110  //--------------------------------------------------------------------------
111  // check inputs
112  //--------------------------------------------------------------------------
113 
114  GrB_Index n, *I = NULL, *Px = NULL, *mem = NULL ;
115  GrB_Vector parent = NULL, gp = NULL, mnp = NULL, ccmn = NULL, ramp = NULL,
116  mask = NULL ;
117  GrB_IndexUnaryOp select_op = NULL ;
118  GrB_Matrix S = NULL ;
119 
120  LG_CLEAR_MSG ;
121  LG_TRY (LAGraph_CheckGraph (G, msg)) ;
122  LG_ASSERT (component != NULL, GrB_NULL_POINTER) ;
123 
124  LG_ASSERT_MSG ((G->kind == LAGraph_ADJACENCY_UNDIRECTED ||
125  (G->kind == LAGraph_ADJACENCY_DIRECTED &&
126  G->is_symmetric_structure == LAGraph_TRUE)),
127  LAGRAPH_SYMMETRIC_STRUCTURE_REQUIRED,
128  "G->A must be known to be symmetric") ;
129 
130  //--------------------------------------------------------------------------
131  // initializations
132  //--------------------------------------------------------------------------
133 
134  // S = structure of G->A
135  LG_TRY (LAGraph_Matrix_Structure (&S, G->A, msg)) ;
136 
137  GRB_TRY (GrB_Matrix_nrows (&n, S)) ;
138  GRB_TRY (GrB_Vector_new (&parent, GrB_UINT64, n)) ; // final result
139  GRB_TRY (GrB_Vector_new (&gp, GrB_UINT64, n)) ; // grandparents
140  GRB_TRY (GrB_Vector_new (&mnp, GrB_UINT64, n)) ; // min neighbor parent
141  GRB_TRY (GrB_Vector_new (&ccmn, GrB_UINT64, n)) ; // cc's min neighbor
142  GRB_TRY (GrB_Vector_new (&mask, GrB_BOOL, n)) ; // various uses
143 
144  LG_TRY (LAGraph_Malloc ((void **) &mem, 3*n, sizeof (GrB_Index), msg)) ;
145  LG_TRY (LAGraph_Malloc ((void **) &Px, n, sizeof (GrB_Index), msg)) ;
146 
147  #if !LAGRAPH_SUITESPARSE
148  // I is not needed for SuiteSparse and remains NULL
149  LG_TRY (LAGraph_Malloc ((void **) &I, n, sizeof (GrB_Index), msg)) ;
150  #endif
151 
152  // parent = 0:n-1, and copy to ramp
153  GRB_TRY (GrB_assign (parent, NULL, NULL, 0, GrB_ALL, n, NULL)) ;
154  GRB_TRY (GrB_apply (parent, NULL, NULL, GrB_ROWINDEX_INT64, parent, 0,
155  NULL)) ;
156  GRB_TRY (GrB_Vector_dup (&ramp, parent)) ;
157 
158  // Px is a non-opaque copy of the parent GrB_Vector
159  GRB_TRY (GrB_Vector_extractTuples (I, Px, &n, parent)) ;
160 
161  GRB_TRY (GrB_IndexUnaryOp_new (&select_op, my_select_func, GrB_BOOL,
162  /* aij: ignored */ GrB_BOOL, /* y: pointer to Px */ GrB_UINT64)) ;
163 
164  GrB_Index nvals ;
165  GRB_TRY (GrB_Matrix_nvals (&nvals, S)) ;
166 
167  //--------------------------------------------------------------------------
168  // find the connected components
169  //--------------------------------------------------------------------------
170 
171  while (nvals > 0)
172  {
173 
174  //----------------------------------------------------------------------
175  // mnp[u] = u's minimum neighbor's parent for all nodes u
176  //----------------------------------------------------------------------
177 
178  // every vertex points to a root vertex at the begining
179  GRB_TRY (GrB_assign (mnp, NULL, NULL, n, GrB_ALL, n, NULL)) ;
180  GRB_TRY (GrB_mxv (mnp, NULL, GrB_MIN_UINT64,
181  GrB_MIN_SECOND_SEMIRING_UINT64, S, parent, NULL)) ;
182 
183  //----------------------------------------------------------------------
184  // find the minimum neighbor
185  //----------------------------------------------------------------------
186 
187  // ccmn[u] = connect component's minimum neighbor | if u is a root
188  // = n | otherwise
189  GRB_TRY (GrB_assign (ccmn, NULL, NULL, n, GrB_ALL, n, NULL)) ;
190  GRB_TRY (Reduce_assign (ccmn, mnp, Px, mem, n)) ;
191 
192  //----------------------------------------------------------------------
193  // parent[u] = ccmn[u] if ccmn[u] != n
194  //----------------------------------------------------------------------
195 
196  // mask = (ccnm != n)
197  GRB_TRY (GrB_apply (mask, NULL, NULL, GrB_NE_UINT64, ccmn, n, NULL)) ;
198  // parent<mask> = ccmn
199  GRB_TRY (GrB_assign (parent, mask, NULL, ccmn, GrB_ALL, n, NULL)) ;
200 
201  //----------------------------------------------------------------------
202  // select new roots
203  //----------------------------------------------------------------------
204 
205  // identify all pairs (u,v) where parent [u] == v and parent [v] == u
206  // and then select the minimum of u, v as the new root;
207  // if (parent [parent [i]] == i) parent [i] = min (parent [i], i)
208 
209  // compute grandparents: gp = parent (parent)
210  GRB_TRY (GrB_Vector_extractTuples (I, Px, &n, parent)) ;
211  GRB_TRY (GrB_extract (gp, NULL, NULL, parent, Px, n, NULL)) ;
212 
213  // mask = (gp == 0:n-1)
214  GRB_TRY (GrB_eWiseMult (mask, NULL, NULL, GrB_EQ_UINT64, gp, ramp,
215  NULL)) ;
216  // parent<mask> = min (parent, ramp)
217  GRB_TRY (GrB_assign (parent, mask, GrB_MIN_UINT64, ramp, GrB_ALL, n,
218  NULL)) ;
219 
220  //----------------------------------------------------------------------
221  // shortcutting: parent [i] = parent [parent [i]] until convergence
222  //----------------------------------------------------------------------
223 
224  bool changing = true ;
225  while (true)
226  {
227  // compute grandparents: gp = parent (parent)
228  GRB_TRY (GrB_Vector_extractTuples (I, Px, &n, parent)) ;
229  GRB_TRY (GrB_extract (gp, NULL, NULL, parent, Px, n, NULL)) ;
230 
231  // changing = or (parent != gp)
232  GRB_TRY (GrB_eWiseMult (mask, NULL, NULL, GrB_NE_UINT64, parent, gp,
233  NULL)) ;
234  GRB_TRY (GrB_reduce (&changing, NULL, GrB_LOR_MONOID_BOOL, mask,
235  NULL)) ;
236  if (!changing) break ;
237 
238  // parent = gp
239  GRB_TRY (GrB_assign (parent, NULL, NULL, gp, GrB_ALL, n, NULL)) ;
240  }
241 
242  //----------------------------------------------------------------------
243  // remove the edges inside each connected component
244  //----------------------------------------------------------------------
245 
246  // This step is the costliest part of this algorithm when dealing with
247  // large matrices.
248  GRB_TRY (GrB_select (S, NULL, NULL, select_op, S, (uint64_t) Px, NULL));
249  GRB_TRY (GrB_Matrix_nvals (&nvals, S)) ;
250  }
251 
252  //--------------------------------------------------------------------------
253  // free workspace and return result
254  //--------------------------------------------------------------------------
255 
256  (*component) = parent ;
257  LG_FREE_WORK ;
258  return (GrB_SUCCESS) ;
259 }
