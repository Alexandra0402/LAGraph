34 #define LG_VECTOR_PRINT(suffix,ctype,gtype,fmt1,fmt2) \
35 int LG_Vector_Print_ ## suffix \
36 ( \
37  GrB_Vector v, LAGraph_PrintLevel pr, FILE *f, char *msg \
38 ) \
39 { \
40  LG_CLEAR_MSG ; \
41  ctype *X = NULL ; \
42  GrB_Index *I = NULL ; \
43  LG_ASSERT (v != NULL, GrB_NULL_POINTER) ; \
44  LG_ASSERT (f != NULL, GrB_NULL_POINTER) ; \
45  int prl = (int) pr ; \
46  if (prl <= 0) return (GrB_SUCCESS) ; \
47  /* get basic properties */ \
48  GrB_Index n, nvals ; \
49  GRB_TRY (GrB_Vector_size (&n, v)) ; \
50  GRB_TRY (GrB_Vector_nvals (&nvals, v)) ; \
51  /* print header line */ \
52  FPRINTF (f, "%s vector: n: %" PRIu64 " entries: %" PRIu64 \
53  "\n", LG_XSTR (gtype), n, nvals) ; \
54  if (prl <= 1) return (GrB_SUCCESS) ; \
55  /* extract tuples */ \
56  LG_TRY (LAGraph_Malloc ((void **) &I, nvals, sizeof (GrB_Index), msg)) ;\
57  LG_TRY (LAGraph_Malloc ((void **) &X, nvals, sizeof (ctype), msg)) ; \
58  GrB_Info info = GrB_Vector_extractTuples (I, X, &nvals, v) ; \
59  LG_ASSERT_MSG (info != GrB_DOMAIN_MISMATCH, \
60  GrB_NOT_IMPLEMENTED, "type not supported") ; \
61  GRB_TRY (info) ; \
62  /* determine the format */ \
63  char *format = (prl <= 3) ? fmt1 : fmt2 ; \
64  bool summary = (prl == 2 || prl == 4) && (nvals > LG_SHORT_LEN) ; \
65  for (int64_t k = 0 ; k < nvals ; k++) \
66  { \
67  /* print the kth tuple */ \
68  GrB_Index i = I [k] ; \
69  ctype x = X [k] ; \
70  FPRINTF (f, " (%" PRIu64 ") ", i) ; \
71  FPRINTF (f, format, x) ; \
72  FPRINTF (f, "\n") ; \
73  if (summary && k > LG_SHORT_LEN) \
74  { \
75  /* quit early if a only a summary is requested */ \
76  FPRINTF (f, " ...\n") ; \
77  break ; \
78  } \
79  } \
80  LG_FREE_WORK ; \
81  return (GrB_SUCCESS) ; \
82 }
