15 // LAGr_Betweenness: Batch algorithm for computing
16 // betweeness centrality, using push-pull optimization.
17 
18 // This is an Advanced algorithm (G->AT is required).
19 
20 // This method computes an approximation of the betweenness algorithm.
21 // ____
22 // \ sigma(s,t | i)
23 // Betweenness centrality = \ ----------------
24 // of node i / sigma(s,t)
25 // /___
26 // s != i != t
27 //
28 // Where sigma(s,t) is the total number of shortest paths from node s to
29 // node t, and sigma(s,t | i) is the total number of shortest paths from
30 // node s to node t that pass through node i.
31 //
32 // Note that the true betweenness centrality requires computing shortest paths
33 // from all nodes s to all nodes t (or all-pairs shortest paths), which can be
34 // expensive to compute. By using a reasonably sized subset of source nodes, an
35 // approximation can be made.
36 //
37 // This method performs simultaneous breadth-first searches of the entire graph
38 // starting at a given set of source nodes. This pass discovers all shortest
39 // paths from the source nodes to all other nodes in the graph. After the BFS
40 // is complete, the number of shortest paths that pass through a given node is
41 // tallied by reversing the traversal. From this, the (approximate) betweenness
42 // centrality is computed.
43 
44 // G->A represents the graph, and G->AT must be present. G->A must be square,
45 // and can be unsymmetric. Self-edges are OK. The values of G->A and G->AT
46 // are ignored; just the structure of two matrices are used.
47 
48 // Each phase uses push-pull direction optimization.
49

78 // LAGr_Betweenness: vertex betweenness-centrality
79 //------------------------------------------------------------------------------
80 
81 int LAGr_Betweenness
82 (
83  // output:
84  GrB_Vector *centrality, // centrality(i): betweeness centrality of i
85  // input:
86  LAGraph_Graph G, // input graph
87  const GrB_Index *sources, // source vertices to compute shortest paths
88  int32_t ns, // number of source vertices
89  char *msg
90 )
91 {
92 
93  //--------------------------------------------------------------------------
94  // check inputs
95  //--------------------------------------------------------------------------
96 
97  LG_CLEAR_MSG ;
98 
99  // Array of BFS search matrices.
100  // S [i] is a sparse matrix that stores the depth at which each vertex is
101  // first seen thus far in each BFS at the current depth i. Each column
102  // corresponds to a BFS traversal starting from a source node.
103  GrB_Matrix *S = NULL ;
104 
105  // Frontier matrix, a sparse matrix.
106  // Stores # of shortest paths to vertices at current BFS depth
107  GrB_Matrix frontier = NULL ;
108 
109  // Paths matrix holds the number of shortest paths for each node and
110  // starting node discovered so far. A dense matrix that is updated with
111  // sparse updates, and also used as a mask.
112  GrB_Matrix paths = NULL ;
113 
114  // Update matrix for betweenness centrality, values for each node for
115  // each starting node. A dense matrix.
116  GrB_Matrix bc_update = NULL ;
117 
118  // Temporary workspace matrix (sparse).
119  GrB_Matrix W = NULL ;
120 
121  GrB_Index n = 0 ; // # nodes in the graph
122 
123  LG_ASSERT (centrality != NULL && sources != NULL, GrB_NULL_POINTER) ;
124  (*centrality) = NULL ;
125  LG_TRY (LAGraph_CheckGraph (G, msg)) ;
126 
127  GrB_Matrix A = G->A ;
128  GrB_Matrix AT ;
129  if (G->kind == LAGraph_ADJACENCY_UNDIRECTED ||
130  G->is_symmetric_structure == LAGraph_TRUE)
131  {
132  // A and A' have the same structure
133  AT = A ;
134  }
135  else
136  {
137  // A and A' differ
138  AT = G->AT ;
139  LG_ASSERT_MSG (AT != NULL, LAGRAPH_NOT_CACHED, "G->AT is required") ;
140  }
141 
142  // =========================================================================
143  // === initializations =====================================================
144  // =========================================================================
145 
146  // Initialize paths and frontier with source notes
147  GRB_TRY (GrB_Matrix_nrows (&n, A)) ;
148  GRB_TRY (GrB_Matrix_new (&paths, GrB_FP64, ns, n)) ;
149  GRB_TRY (GrB_Matrix_new (&frontier, GrB_FP64, ns, n)) ;
150  #if LAGRAPH_SUITESPARSE
151  GRB_TRY (GxB_set (paths, GxB_SPARSITY_CONTROL, GxB_BITMAP + GxB_FULL)) ;
152  #endif
153  for (GrB_Index i = 0 ; i < ns ; i++)
154  {
155  // paths (i,s(i)) = 1
156  // frontier (i,s(i)) = 1
157  double one = 1 ;
158  GrB_Index src = sources [i] ;
159  LG_ASSERT_MSG (src < n, GrB_INVALID_INDEX, "invalid source node") ;
160  GRB_TRY (GrB_Matrix_setElement (paths, one, i, src)) ;
161  GRB_TRY (GrB_Matrix_setElement (frontier, one, i, src)) ;
162  }
163 
164  // Initial frontier: frontier<!paths>= frontier*A
165  GRB_TRY (GrB_mxm (frontier, paths, NULL, LAGraph_plus_first_fp64,
166  frontier, A, GrB_DESC_RSC)) ;
167 
168  // Allocate memory for the array of S matrices
169  LG_TRY (LAGraph_Malloc ((void **) &S, n+1, sizeof (GrB_Matrix), msg)) ;
170  S [0] = NULL ;
171 
172  // =========================================================================
173  // === Breadth-first search stage ==========================================
174  // =========================================================================
175 
176  bool last_was_pull = false ;
177  GrB_Index frontier_size, last_frontier_size = 0 ;
178  GRB_TRY (GrB_Matrix_nvals (&frontier_size, frontier)) ;
179 
180  int64_t depth ;
181  for (depth = 0 ; frontier_size > 0 && depth < n ; depth++)
182  {
183 
184  //----------------------------------------------------------------------
185  // S [depth] = structure of frontier
186  //----------------------------------------------------------------------
187 
188  S [depth+1] = NULL ;
189  LG_TRY (LAGraph_Matrix_Structure (&(S [depth]), frontier, msg)) ;
190 
191  //----------------------------------------------------------------------
192  // Accumulate path counts: paths += frontier
193  //----------------------------------------------------------------------
194 
195  GRB_TRY (GrB_assign (paths, NULL, GrB_PLUS_FP64, frontier, GrB_ALL, ns,
196  GrB_ALL, n, NULL)) ;
197 
198  //----------------------------------------------------------------------
199  // Update frontier: frontier<!paths> = frontier*A
200  //----------------------------------------------------------------------
201 
202  // pull if frontier is more than 10% dense,
203  // or > 6% dense and last step was pull
204  double frontier_density = ((double) frontier_size) / (double) (ns*n) ;
205  bool do_pull = frontier_density > (last_was_pull ? 0.06 : 0.10 ) ;
206 
207  if (do_pull)
208  {
209  // frontier<!paths> = frontier*AT'
210  #if LAGRAPH_SUITESPARSE
211  GRB_TRY (GxB_set (frontier, GxB_SPARSITY_CONTROL, GxB_BITMAP)) ;
212  #endif
213  GRB_TRY (GrB_mxm (frontier, paths, NULL, LAGraph_plus_first_fp64,
214  frontier, AT, GrB_DESC_RSCT1)) ;
215  }
216  else // push
217  {
218  // frontier<!paths> = frontier*A
219  #if LAGRAPH_SUITESPARSE
220  GRB_TRY (GxB_set (frontier, GxB_SPARSITY_CONTROL, GxB_SPARSE)) ;
221  #endif
222  GRB_TRY (GrB_mxm (frontier, paths, NULL, LAGraph_plus_first_fp64,
223  frontier, A, GrB_DESC_RSC)) ;
224  }
225 
226  //----------------------------------------------------------------------
227  // Get size of current frontier: frontier_size = nvals(frontier)
228  //----------------------------------------------------------------------
229 
230  last_frontier_size = frontier_size ;
231  last_was_pull = do_pull ;
232  GRB_TRY (GrB_Matrix_nvals (&frontier_size, frontier)) ;
233  }
234 
235  GRB_TRY (GrB_free (&frontier)) ;
236 
237  // =========================================================================
238  // === Betweenness centrality computation phase ============================
239  // =========================================================================
240 
241  // bc_update = ones (ns, n) ; a full matrix (and stays full)
242  GRB_TRY (GrB_Matrix_new (&bc_update, GrB_FP64, ns, n)) ;
243  GRB_TRY (GrB_assign (bc_update, NULL, NULL, 1, GrB_ALL, ns, GrB_ALL, n,
244  NULL)) ;
245  // W: empty ns-by-n array, as workspace
246  GRB_TRY (GrB_Matrix_new (&W, GrB_FP64, ns, n)) ;
247 
248  // Backtrack through the BFS and compute centrality updates for each vertex
249  for (int64_t i = depth-1 ; i > 0 ; i--)
250  {
251 
252  //----------------------------------------------------------------------
253  // W<S[i]> = bc_update ./ paths
254  //----------------------------------------------------------------------
255 
256  // Add contributions by successors and mask with that level's frontier
257  GRB_TRY (GrB_eWiseMult (W, S [i], NULL, GrB_DIV_FP64, bc_update, paths,
258  GrB_DESC_RS)) ;
259 
260  //----------------------------------------------------------------------
261  // W<S[i−1]> = W * A'
262  //----------------------------------------------------------------------
263 
264  // pull if W is more than 10% dense and nnz(W)/nnz(S[i-1]) > 1
265  // or if W is more than 1% dense and nnz(W)/nnz(S[i-1]) > 10
266  GrB_Index wsize, ssize ;
267  GrB_Matrix_nvals (&wsize, W) ;
268  GrB_Matrix_nvals (&ssize, S [i-1]) ;
269  double w_density = ((double) wsize) / ((double) (ns*n)) ;
270  double w_to_s_ratio = ((double) wsize) / ((double) ssize) ;
271  bool do_pull = (w_density > 0.1 && w_to_s_ratio > 1.) ||
272  (w_density > 0.01 && w_to_s_ratio > 10.) ;
273 
274  if (do_pull)
275  {
276  // W<S[i−1]> = W * A'
277  #if LAGRAPH_SUITESPARSE
278  GRB_TRY (GxB_set (W, GxB_SPARSITY_CONTROL, GxB_BITMAP)) ;
279  #endif
280  GRB_TRY (GrB_mxm (W, S [i-1], NULL, LAGraph_plus_first_fp64, W, A,
281  GrB_DESC_RST1)) ;
282  }
283  else // push
284  {
285  // W<S[i−1]> = W * AT
286  #if LAGRAPH_SUITESPARSE
287  GRB_TRY (GxB_set (W, GxB_SPARSITY_CONTROL, GxB_SPARSE)) ;
288  #endif
289  GRB_TRY (GrB_mxm (W, S [i-1], NULL, LAGraph_plus_first_fp64, W, AT,
290  GrB_DESC_RS)) ;
291  }
292 
293  //----------------------------------------------------------------------
294  // bc_update += W .* paths
295  //----------------------------------------------------------------------
296 
297  GRB_TRY (GrB_eWiseMult (bc_update, NULL, GrB_PLUS_FP64, GrB_TIMES_FP64,
298  W, paths, NULL)) ;
299  }
300 
301  // =========================================================================
302  // === finalize the centrality =============================================
303  // =========================================================================
304 
305  // Initialize the centrality array with -ns to avoid counting
306  // zero length paths
307  GRB_TRY (GrB_Vector_new (centrality, GrB_FP64, n)) ;
308  GRB_TRY (GrB_assign (*centrality, NULL, NULL, -ns, GrB_ALL, n, NULL)) ;
309 
310  // centrality (i) += sum (bc_update (:,i)) for all nodes i
311  GRB_TRY (GrB_reduce (*centrality, NULL, GrB_PLUS_FP64, GrB_PLUS_MONOID_FP64,
312  bc_update, GrB_DESC_T0)) ;
313 
314  LG_FREE_WORK ;
315  return (GrB_SUCCESS) ;
316 }
