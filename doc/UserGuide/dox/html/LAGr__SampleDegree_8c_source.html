14 // LAGr_SampleDegree computes estimates of the mean and median of the
15 // row or column degree of a graph.
16 
17 #define LG_FREE_ALL LAGraph_Free ((void **) &samples, NULL) ;
18 
19 #include "LG_internal.h"
20 
21 int LAGr_SampleDegree
22 (
23  // output:
24  double *sample_mean, // sampled mean degree
25  double *sample_median, // sampled median degree
26  // input:
27  const LAGraph_Graph G, // graph of n nodes
28  bool byout, // if true, sample G->out_degree, else G->in_degree
29  int64_t nsamples, // number of samples
30  uint64_t seed, // random number seed
31  char *msg
32 )
33 {
34 
35  //--------------------------------------------------------------------------
36  // check inputs
37  //--------------------------------------------------------------------------
38 
39  LG_CLEAR_MSG ;
40  int64_t *samples = NULL ;
41  LG_ASSERT (sample_mean != NULL, GrB_NULL_POINTER) ;
42  LG_ASSERT (sample_median != NULL, GrB_NULL_POINTER) ;
43  nsamples = LAGRAPH_MAX (nsamples, 1) ;
44  LG_TRY (LAGraph_CheckGraph (G, msg)) ;
45 
46  GrB_Vector Degree ;
47 
48  if (G->kind == LAGraph_ADJACENCY_UNDIRECTED ||
49  (G->kind == LAGraph_ADJACENCY_DIRECTED &&
50  G->is_symmetric_structure == LAGraph_TRUE))
51  {
52  // the structure of A is known to be symmetric
53  Degree = G->out_degree ;
54  }
55  else
56  {
57  // A is not known to be symmetric
58  Degree = (byout) ? G->out_degree : G->in_degree ;
59  }
60 
61  LG_ASSERT_MSG (Degree != NULL, LAGRAPH_NOT_CACHED, "degree unknown") ;
62 
63  //--------------------------------------------------------------------------
64  // allocate workspace
65  //--------------------------------------------------------------------------
66 
67  LG_TRY (LAGraph_Malloc ((void **) &samples, nsamples, sizeof (int64_t),
68  msg)) ;
69 
70  //--------------------------------------------------------------------------
71  // pick nsamples nodes at random and determine their degree
72  //--------------------------------------------------------------------------
73 
74  // See also the hashed sampling method in LG_CC_FastSV6, which computes a
75  // fast estimate of the mode of an integer vector. This method does not
76  // require a hash table. However, the mode estimator in LG_CC_FastSV6
77  // would be a good candidate to add as an LAGraph_SampleMode utility
78  // function.
79 
80  GrB_Index n ;
81  GRB_TRY (GrB_Vector_size (&n, Degree)) ;
82 
83  int64_t dsum = 0 ;
84  for (int k = 0 ; k < nsamples ; k++)
85  {
86  uint64_t result = LG_Random60 (&seed) ;
87  int64_t i = result % n ;
88  // d = Degree (i)
89  int64_t d ;
90  GRB_TRY (GrB_Vector_extractElement (&d, Degree, i)) ;
91  samples [k] = d ;
92  dsum += d ;
93  }
94 
95  // find the mean degree
96  (*sample_mean) = ((double) dsum) / nsamples ;
97 
98  // find the median degree
99  LG_qsort_1a (samples, nsamples) ;
100  (*sample_median) = (double) samples [nsamples/2] ;
101 
102  //--------------------------------------------------------------------------
103  // free workspace and return result
104  //--------------------------------------------------------------------------
105 
106  LG_FREE_ALL ;
107  return (GrB_SUCCESS) ;
108 }
