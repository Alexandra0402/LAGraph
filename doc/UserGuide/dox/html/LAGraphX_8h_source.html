81 // binary file I/O
82 //****************************************************************************
83 
84 // The LAGraph *.lagraph file consists of an ASCII JSON header, followed by
85 // one or more serialized "blobs" created by GrB_Matrix_serialize (or
86 // GxB_Matrix_serialize if using SuiteSparse:GraphBLAS). The file can only be
87 // read back into LAGraph when using the same GraphBLAS library used to create
88 // it.
89 
90 // To create a binary file containing one or more GrB_Matrix objects, the user
91 // application must first open the file f, create the ascii JSON header with
92 // LAGraph_SWrite_Header*, and then write one or more binary serialized
93 // GrB_Matrix blobs from using LAGraph_SWrite_Matrix.
94 
95 // Example:
96 
97 /*
98  // serialize the matrices A (of type GrB_FP64) and B (of type GrB_BOOL)
99  void *Ablob, *Bblob ;
100  GrB_Index Ablob_size, Bblob_size ;
101  GxB_Matrix_serialize (&Ablob, &Ablob_size, A, NULL) ;
102  GxB_Matrix_serialize (&Bblob, &Bblob_size, B, NULL) ;
103 
104  // open the file and write the JSON header
105  FILE *f = fopen ("mymatrices.lagraph", "w") ;
106  LAGraph_SWrite_HeaderStart (f, "mystuff", msg) ;
107  LAGraph_SWrite_HeaderItem (f, LAGraph_matrix_kind, "A", "double", 0,
108  Ablob_size, msg) ;
109  LAGraph_SWrite_HeaderItem (f, LAGraph_matrix_kind, "B", "bool", 0,
110  Bblob_size, msg) ;
111  LAGraph_SWrite_HeaderEnd (f, msg) ;
112 
113  // write the matrices in binary
114  LAGraph_SWrite_Item (f, Ablob, Ablob_size, msg) ;
115  LAGraph_SWrite_Item (f, Bblob, Bblob_size, msg) ;
116 
117  fclose (f) ;
118 */
119

123  LAGraph_matrix_kind = 0, // a serialized GrB_Matrix
124  LAGraph_vector_kind = 1, // a serialized GrB_Vector (SS:GrB only)
125  LAGraph_text_kind = 2, // text (char *), possibly compressed
126 }
