14 // If p is non-NULL on input, it points to a previously allocated object of
15 // size at least nitems_old * size_of_item. The object is reallocated to be of
16 // size at least nitems_new * size_of_item. If p is NULL on input, then a new
17 // object of that size is allocated. On success, a pointer to the new object
18 // is returned, and ok is returned as true. If the allocation fails, ok is set
19 // to false and a pointer to the old (unmodified) object is returned.
20 
21 // Usage:
22 
23 // int status = LAGraph_Realloc (&p, nitems_new, nitems_old, size_of_item, msg)
24 // if (status == GrB_SUCCESS)
25 // {
26 // p points to a block of at least nitems_new*size_of_item bytes and
27 // the first part, of size min(nitems_new,nitems_old)*size_of_item,
28 // has the same content as the old memory block if it was present.
29 // }
30 // else
31 // {
32 // p points to the old block, unchanged. This case never occurs if
33 // nitems_new < nitems_old.
34 // }
35 
36 #include "LG_internal.h"
37 
38 int LAGraph_Realloc
39 (
40  // input/output:
41  void **p, // old block to reallocate
42  // input:
43  size_t nitems_new, // new number of items in the object
44  size_t nitems_old, // old number of items in the object
45  size_t size_of_item, // size of each item
46  char *msg
47 )
48 {
49 
50  //--------------------------------------------------------------------------
51  // check inputs
52  //--------------------------------------------------------------------------
53 
54  LG_CLEAR_MSG ;
55  LG_ASSERT (p != NULL, GrB_NULL_POINTER) ;
56 
57  //--------------------------------------------------------------------------
58  // malloc a new block if p is NULL on input
59  //--------------------------------------------------------------------------
60 
61  if ((*p) == NULL)
62  {
63  LG_TRY (LAGraph_Malloc (p, nitems_new, size_of_item, msg)) ;
64  return (GrB_SUCCESS) ;
65  }
66 
67  //--------------------------------------------------------------------------
68  // check inputs
69  //--------------------------------------------------------------------------
70 
71  // make sure at least one item is allocated
72  nitems_old = LAGRAPH_MAX (1, nitems_old) ;
73  nitems_new = LAGRAPH_MAX (1, nitems_new) ;
74 
75  // make sure at least one byte is allocated
76  size_of_item = LAGRAPH_MAX (1, size_of_item) ;
77 
78  size_t newsize, oldsize ;
79  bool ok = LG_Multiply_size_t (&newsize, nitems_new, size_of_item)
80  && LG_Multiply_size_t (&oldsize, nitems_old, size_of_item) ;
81 
82  if (!ok || nitems_new > GrB_INDEX_MAX || size_of_item > GrB_INDEX_MAX)
83  {
84  // overflow
85  return (GrB_OUT_OF_MEMORY) ;
86  }
87 
88  //--------------------------------------------------------------------------
89  // reallocate an existing block to accommodate the change in # of items
90  //--------------------------------------------------------------------------
91 
92  //--------------------------------------------------------------------------
93  // check for quick return
94  //--------------------------------------------------------------------------
95 
96  if (newsize == oldsize)
97  {
98  // If the block does not change, or is shrinking but only by a small
99  // amount, or is growing but still fits inside the existing block,
100  // then leave the block as-is.
101  return (GrB_SUCCESS) ;
102  }
103 
104  //--------------------------------------------------------------------------
105  // reallocate the memory, or use malloc/memcpy/free
106  //--------------------------------------------------------------------------
107 
108  void *pnew = NULL ;
109 
110  if (LAGraph_Realloc_function == NULL)
111  {
112 
113  //----------------------------------------------------------------------
114  // use malloc/memcpy/free
115  //----------------------------------------------------------------------
116 
117  // allocate the new space
118  LG_TRY (LAGraph_Malloc (&pnew, nitems_new, size_of_item, msg)) ;
119  // copy over the data from the old block to the new block
120  // copy from the old to the new space
121  memcpy (pnew, *p, LAGRAPH_MIN (oldsize, newsize)) ;
122  // free the old space
123  LG_TRY (LAGraph_Free (p, msg)) ;
124  (*p) = pnew ;
125 
126  }
127  else
128  {
129 
130  //----------------------------------------------------------------------
131  // use realloc
132  //----------------------------------------------------------------------
133 
134  pnew = LAGraph_Realloc_function (*p, newsize) ;
135  if (pnew == NULL) return (GrB_OUT_OF_MEMORY) ;
136  (*p) = pnew ;
137  }
138 
139  return (GrB_SUCCESS) ;
140 }
