16 int LAGraph_Calloc
17 (
18  // output:
19  void **p, // pointer to allocated block of memory
20  // input:
21  size_t nitems, // number of items
22  size_t size_of_item, // size of each item
23  char *msg
24 )
25 {
26  // check inputs
27  LG_CLEAR_MSG ;
28  LG_ASSERT (p != NULL, GrB_NULL_POINTER) ;
29  (*p) = NULL ;
30 
31  // make sure at least one item is allocated
32  nitems = LAGRAPH_MAX (1, nitems) ;
33 
34  // make sure at least one byte is allocated
35  size_of_item = LAGRAPH_MAX (1, size_of_item) ;
36 
37  // compute the size and check for integer overflow
38  size_t size ;
39  bool ok = LG_Multiply_size_t (&size, nitems, size_of_item) ;
40  if (!ok || nitems > GrB_INDEX_MAX || size_of_item > GrB_INDEX_MAX)
41  {
42  // overflow
43  return (GrB_OUT_OF_MEMORY) ;
44  }
45 
46  if (LAGraph_Calloc_function == NULL)
47  {
48  // calloc function not available; use malloc and memset
49  LG_TRY (LAGraph_Malloc (p, nitems, size_of_item, msg)) ;
50  memset (*p, 0, size) ;
51  return (GrB_SUCCESS) ;
52  }
53 
54  // use the calloc function
55  (*p) = LAGraph_Calloc_function (nitems, size_of_item) ;
56  return (((*p) == NULL) ? GrB_OUT_OF_MEMORY : GrB_SUCCESS) ;
57 }
