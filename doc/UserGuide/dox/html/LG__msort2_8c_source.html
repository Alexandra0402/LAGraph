47 //------------------------------------------------------------------------------
48 // LG_msort_2b_binary_search: binary search for the pivot
49 //------------------------------------------------------------------------------
50 
51 // The Pivot value is Y [pivot], and a binary search for the Pivot is made in
52 // the array X [p_pstart...p_end-1], which is sorted in non-decreasing order on
53 // input. The return value is pleft, where
54 //
55 // X [p_start ... pleft-1] <= Pivot and
56 // X [pleft ... p_end-1] >= Pivot holds.
57 //
58 // pleft is returned in the range p_start to p_end. If pleft is p_start, then
59 // the Pivot is smaller than all entries in X [p_start...p_end-1], and the left
60 // list X [p_start...pleft-1] is empty. If pleft is p_end, then the Pivot is
61 // larger than all entries in X [p_start...p_end-1], and the right list X
62 // [pleft...p_end-1] is empty.
63 
64 static int64_t LG_msort_2b_binary_search // return pleft
65 (
66  const int64_t *LG_RESTRICT Y_0, // Pivot is Y [pivot]
67  const int64_t *LG_RESTRICT Y_1,
68  const int64_t pivot,
69  const int64_t *LG_RESTRICT X_0, // search in X [p_start..p_end_-1]
70  const int64_t *LG_RESTRICT X_1,
71  const int64_t p_start,
72  const int64_t p_end
73 )
74 {
75 
76  //--------------------------------------------------------------------------
77  // find where the Pivot appears in X
78  //--------------------------------------------------------------------------
79 
80  // binary search of X [p_start...p_end-1] for the Pivot
81  int64_t pleft = p_start ;
82  int64_t pright = p_end - 1 ;
83  while (pleft < pright)
84  {
85  int64_t pmiddle = (pleft + pright) >> 1 ;
86  // less = (X [pmiddle] < Pivot)
87  bool less = LG_lt_2 (X_0, X_1, pmiddle,
88  Y_0, Y_1, pivot) ;
89  pleft = less ? (pmiddle+1) : pleft ;
90  pright = less ? pright : pmiddle ;
91  }
92 
93  // binary search is narrowed down to a single item
94  // or it has found the list is empty:
95  ASSERT (pleft == pright || pleft == pright + 1) ;
96 
97  // If found is true then X [pleft == pright] == Pivot. If duplicates
98  // appear then X [pleft] is any one of the entries equal to the Pivot
99  // in the list. If found is false then
100  // X [p_start ... pleft-1] < Pivot and
101  // X [pleft+1 ... p_end-1] > Pivot holds.
102  // The value X [pleft] may be either < or > Pivot.
103  bool found = (pleft == pright) && LG_eq_2 (X_0, X_1, pleft,
104  Y_0, Y_1, pivot) ;
105 
106  // Modify pleft and pright:
107  if (!found && (pleft == pright))
108  {
109  if (LG_lt_2 (X_0, X_1, pleft,
110  Y_0, Y_1, pivot))
111  {
112  pleft++ ;
113  }
114  else
115  {
116 // pright++ ; // (not needed)
117  }
118  }
119 
120  //--------------------------------------------------------------------------
121  // return result
122  //--------------------------------------------------------------------------
123 
124  // If found is false then
125  // X [p_start ... pleft-1] < Pivot and
126  // X [pleft ... p_end-1] > Pivot holds,
127  // and pleft-1 == pright
128 
129  // If X has no duplicates, then whether or not Pivot is found,
130  // X [p_start ... pleft-1] < Pivot and
131  // X [pleft ... p_end-1] >= Pivot holds.
132 
133  // If X has duplicates, then whether or not Pivot is found,
134  // X [p_start ... pleft-1] <= Pivot and
135  // X [pleft ... p_end-1] >= Pivot holds.
136 
137  return (pleft) ;
138 }

144 // Recursively constructs ntasks tasks to merge two arrays, Left and Right,
145 // into Sresult, where Left is L [pL_start...pL_end-1], Right is R
146 // [pR_start...pR_end-1], and Sresult is S [pS_start...pS_start+total_work-1],
147 // and where total_work is the total size of Left and Right.
148 //
149 // Task tid will merge L [L_task [tid] ... L_task [tid] + L_len [tid] - 1] and
150 // R [R_task [tid] ... R_task [tid] + R_len [tid] -1] into the merged output
151 // array S [S_task [tid] ... ]. The task tids created are t0 to
152 // t0+ntasks-1.
153 
154 void LG_msort_2b_create_merge_tasks
155 (
156  // output:
157  int64_t *LG_RESTRICT L_task, // L_task [t0...t0+ntasks-1] computed
158  int64_t *LG_RESTRICT L_len, // L_len [t0...t0+ntasks-1] computed
159  int64_t *LG_RESTRICT R_task, // R_task [t0...t0+ntasks-1] computed
160  int64_t *LG_RESTRICT R_len, // R_len [t0...t0+ntasks-1] computed
161  int64_t *LG_RESTRICT S_task, // S_task [t0...t0+ntasks-1] computed
162  // input:
163  const int t0, // first task tid to create
164  const int ntasks, // # of tasks to create
165  const int64_t pS_start, // merge into S [pS_start...]
166  const int64_t *LG_RESTRICT L_0, // Left = L [pL_start...pL_end-1]
167  const int64_t *LG_RESTRICT L_1,
168  const int64_t pL_start,
169  const int64_t pL_end,
170  const int64_t *LG_RESTRICT R_0, // Right = R [pR_start...pR_end-1]
171  const int64_t *LG_RESTRICT R_1,
172  const int64_t pR_start,
173  const int64_t pR_end
174 )
175 {
176 
177  //--------------------------------------------------------------------------
178  // get problem size
179  //--------------------------------------------------------------------------
180 
181  int64_t nleft = pL_end - pL_start ; // size of Left array
182  int64_t nright = pR_end - pR_start ; // size of Right array
183  int64_t total_work = nleft + nright ; // total work to do
184  ASSERT (ntasks >= 1) ;
185  ASSERT (total_work > 0) ;
186 
187  //--------------------------------------------------------------------------
188  // create the tasks
189  //--------------------------------------------------------------------------
190 
191  if (ntasks == 1)
192  {
193 
194  //----------------------------------------------------------------------
195  // a single task will merge all of Left and Right into Sresult
196  //----------------------------------------------------------------------
197 
198  L_task [t0] = pL_start ; L_len [t0] = nleft ;
199  R_task [t0] = pR_start ; R_len [t0] = nright ;
200  S_task [t0] = pS_start ;
201 
202  }
203  else
204  {
205 
206  //----------------------------------------------------------------------
207  // partition the Left and Right arrays for multiple merge tasks
208  //----------------------------------------------------------------------
209 
210  int64_t pleft, pright ;
211  if (nleft >= nright)
212  {
213  // split Left in half, and search for its pivot in Right
214  pleft = (pL_end + pL_start) >> 1 ;
215  pright = LG_msort_2b_binary_search (
216  L_0, L_1, pleft,
217  R_0, R_1, pR_start, pR_end) ;
218  }
219  else
220  {
221  // split Right in half, and search for its pivot in Left
222  pright = (pR_end + pR_start) >> 1 ;
223  pleft = LG_msort_2b_binary_search (
224  R_0, R_1, pright,
225  L_0, L_1, pL_start, pL_end) ;
226  }
227 
228  //----------------------------------------------------------------------
229  // partition the tasks according to the work of each partition
230  //----------------------------------------------------------------------
231 
232  // work0 is the total work in the first partition
233  int64_t work0 = (pleft - pL_start) + (pright - pR_start) ;
234  int ntasks0 = (int) round ((double) ntasks *
235  (((double) work0) / ((double) total_work))) ;
236 
237  // ensure at least one task is assigned to each partition
238  ntasks0 = LAGRAPH_MAX (ntasks0, 1) ;
239  ntasks0 = LAGRAPH_MIN (ntasks0, ntasks-1) ;
240  int ntasks1 = ntasks - ntasks0 ;
241 
242  //----------------------------------------------------------------------
243  // assign ntasks0 to the first half
244  //----------------------------------------------------------------------
245 
246  // ntasks0 tasks merge L [pL_start...pleft-1] and R [pR_start..pright-1]
247  // into the result S [pS_start...work0-1].
248 
249  LG_msort_2b_create_merge_tasks (
250  L_task, L_len, R_task, R_len, S_task, t0, ntasks0, pS_start,
251  L_0, L_1, pL_start, pleft,
252  R_0, R_1, pR_start, pright) ;
253 
254  //----------------------------------------------------------------------
255  // assign ntasks1 to the second half
256  //----------------------------------------------------------------------
257 
258  // ntasks1 tasks merge L [pleft...pL_end-1] and R [pright...pR_end-1]
259  // into the result S [pS_start+work0...pS_start+total_work].
260 
261  int t1 = t0 + ntasks0 ; // first task id of the second set of tasks
262  int64_t pS_start1 = pS_start + work0 ; // 2nd set starts here in S
263  LG_msort_2b_create_merge_tasks (
264  L_task, L_len, R_task, R_len, S_task, t1, ntasks1, pS_start1,
265  L_0, L_1, pleft, pL_end,
266  R_0, R_1, pright, pR_end) ;
267  }
268 }

271 // LG_msort_2b_merge: merge two sorted lists via a single thread
272 //------------------------------------------------------------------------------
273 
274 // merge Left [0..nleft-1] and Right [0..nright-1] into S [0..nleft+nright-1] */
275 
276 static void LG_msort_2b_merge
277 (
278  int64_t *LG_RESTRICT S_0, // output of length nleft + nright
279  int64_t *LG_RESTRICT S_1,
280  const int64_t *LG_RESTRICT Left_0, // left input of length nleft
281  const int64_t *LG_RESTRICT Left_1,
282  const int64_t nleft,
283  const int64_t *LG_RESTRICT Right_0, // right input of length nright
284  const int64_t *LG_RESTRICT Right_1,
285  const int64_t nright
286 )
287 {
288  int64_t p, pleft, pright ;
289 
290  // merge the two inputs, Left and Right, while both inputs exist
291  for (p = 0, pleft = 0, pright = 0 ; pleft < nleft && pright < nright ; p++)
292  {
293  if (LG_lt_2 (Left_0, Left_1, pleft,
294  Right_0, Right_1, pright))
295  {
296  // S [p] = Left [pleft++]
297  S_0 [p] = Left_0 [pleft] ;
298  S_1 [p] = Left_1 [pleft] ;
299  pleft++ ;
300  }
301  else
302  {
303  // S [p] = Right [pright++]
304  S_0 [p] = Right_0 [pright] ;
305  S_1 [p] = Right_1 [pright] ;
306  pright++ ;
307  }
308  }
309 
310  // either input is exhausted; copy the remaining list into S
311  if (pleft < nleft)
312  {
313  int64_t nremaining = (nleft - pleft) ;
314  memcpy (S_0 + p, Left_0 + pleft, nremaining * sizeof (int64_t)) ;
315  memcpy (S_1 + p, Left_1 + pleft, nremaining * sizeof (int64_t)) ;
316  }
317  else if (pright < nright)
318  {
319  int64_t nremaining = (nright - pright) ;
320  memcpy (S_0 + p, Right_0 + pright, nremaining * sizeof (int64_t)) ;
321  memcpy (S_1 + p, Right_1 + pright, nremaining * sizeof (int64_t)) ;
322  }
323 }

329 int LG_msort2
330 (
331  // input/output:
332  int64_t *A_0, // size n array
333  int64_t *A_1, // size n array
334  // input:
335  const int64_t n,
336  char *msg
337 )
338 {
339 
340  //--------------------------------------------------------------------------
341  // check inputs
342  //--------------------------------------------------------------------------
343 
344  LG_CLEAR_MSG ;
345  int64_t *LG_RESTRICT W = NULL ;
346  LG_ASSERT (A_0 != NULL, GrB_NULL_POINTER) ;
347  LG_ASSERT (A_1 != NULL, GrB_NULL_POINTER) ;
348 
349  //--------------------------------------------------------------------------
350  // handle small problems with a single thread
351  //--------------------------------------------------------------------------
352 
353  int nthreads = LG_nthreads_outer * LG_nthreads_inner ; // # threads to use
354  if (nthreads <= 1 || n <= LG_BASECASE)
355  {
356  // sequential quicksort
357  LG_qsort_2 (A_0, A_1, n) ;
358  return (GrB_SUCCESS) ;
359  }
360 
361  //--------------------------------------------------------------------------
362  // determine # of tasks
363  //--------------------------------------------------------------------------
364 
365  // determine the number of levels to create, which must always be an
366  // even number. The # of levels is chosen to ensure that the # of leaves
367  // of the task tree is between 4*nthreads and 16*nthreads.
368 
369  // 2 to 4 threads: 4 levels, 16 qsort leaves
370  // 5 to 16 threads: 6 levels, 64 qsort leaves
371  // 17 to 64 threads: 8 levels, 256 qsort leaves
372  // 65 to 256 threads: 10 levels, 1024 qsort leaves
373  // 256 to 1024 threads: 12 levels, 4096 qsort leaves
374  // ...
375 
376  int k = (int) (2 + 2 * ceil (log2 ((double) nthreads) / 2)) ;
377  int ntasks = 1 << k ;
378 
379  //--------------------------------------------------------------------------
380  // allocate workspace
381  //--------------------------------------------------------------------------
382 
383  LG_TRY (LAGraph_Malloc ((void **) &W, 2*n + 6*ntasks + 1, sizeof (int64_t),
384  msg)) ;
385 
386  int64_t *T = W ;
387  int64_t *LG_RESTRICT W_0 = T ; T += n ;
388  int64_t *LG_RESTRICT W_1 = T ; T += n ;
389  int64_t *LG_RESTRICT L_task = T ; T += ntasks ;
390  int64_t *LG_RESTRICT L_len = T ; T += ntasks ;
391  int64_t *LG_RESTRICT R_task = T ; T += ntasks ;
392  int64_t *LG_RESTRICT R_len = T ; T += ntasks ;
393  int64_t *LG_RESTRICT S_task = T ; T += ntasks ;
394  int64_t *LG_RESTRICT Slice = T ; T += (ntasks+1) ;
395 
396  //--------------------------------------------------------------------------
397  // partition and sort the leaves
398  //--------------------------------------------------------------------------
399 
400  LG_eslice (Slice, n, ntasks) ;
401  int tid ;
402  #pragma omp parallel for num_threads(nthreads) schedule(dynamic,1)
403  for (tid = 0 ; tid < ntasks ; tid++)
404  {
405  int64_t leaf = Slice [tid] ;
406  int64_t leafsize = Slice [tid+1] - leaf ;
407  LG_qsort_2 (A_0 + leaf, A_1 + leaf, leafsize) ;
408  }
409 
410  //--------------------------------------------------------------------------
411  // merge each level
412  //--------------------------------------------------------------------------
413 
414  int nt = 1 ;
415  for ( ; k >= 2 ; k -= 2)
416  {
417 
418  //----------------------------------------------------------------------
419  // merge level k into level k-1, from A into W
420  //----------------------------------------------------------------------
421 
422  // FUTURE: skip k and k-1 for each group of 4 sublists of A if they are
423  // already sorted with respect to each other.
424 
425  // this could be done in parallel if ntasks was large
426  for (int tid = 0 ; tid < ntasks ; tid += 2*nt)
427  {
428  // create 2*nt tasks to merge two A sublists into one W sublist
429  LG_msort_2b_create_merge_tasks (
430  L_task, L_len, R_task, R_len, S_task, tid, 2*nt, Slice [tid],
431  A_0, A_1, Slice [tid], Slice [tid+nt],
432  A_0, A_1, Slice [tid+nt], Slice [tid+2*nt]) ;
433  }
434 
435  #pragma omp parallel for num_threads(nthreads) schedule(dynamic,1)
436  for (tid = 0 ; tid < ntasks ; tid++)
437  {
438  // merge A [pL...pL+nL-1] and A [pR...pR+nR-1] into W [pS..]
439  int64_t pL = L_task [tid], nL = L_len [tid] ;
440  int64_t pR = R_task [tid], nR = R_len [tid] ;
441  int64_t pS = S_task [tid] ;
442 
443  LG_msort_2b_merge (
444  W_0 + pS, W_1 + pS,
445  A_0 + pL, A_1 + pL, nL,
446  A_0 + pR, A_1 + pR, nR) ;
447  }
448  nt = 2*nt ;
449 
450  //----------------------------------------------------------------------
451  // merge level k-1 into level k-2, from W into A
452  //----------------------------------------------------------------------
453 
454  // this could be done in parallel if ntasks was large
455  for (int tid = 0 ; tid < ntasks ; tid += 2*nt)
456  {
457  // create 2*nt tasks to merge two W sublists into one A sublist
458  LG_msort_2b_create_merge_tasks (
459  L_task, L_len, R_task, R_len, S_task, tid, 2*nt, Slice [tid],
460  W_0, W_1, Slice [tid], Slice [tid+nt],
461  W_0, W_1, Slice [tid+nt], Slice [tid+2*nt]) ;
462  }
463 
464  #pragma omp parallel for num_threads(nthreads) schedule(dynamic,1)
465  for (tid = 0 ; tid < ntasks ; tid++)
466  {
467  // merge A [pL...pL+nL-1] and A [pR...pR+nR-1] into W [pS..]
468  int64_t pL = L_task [tid], nL = L_len [tid] ;
469  int64_t pR = R_task [tid], nR = R_len [tid] ;
470  int64_t pS = S_task [tid] ;
471  LG_msort_2b_merge (
472  A_0 + pS, A_1 + pS,
473  W_0 + pL, W_1 + pL, nL,
474  W_0 + pR, W_1 + pR, nR) ;
475  }
476  nt = 2*nt ;
477  }
478 
479  //--------------------------------------------------------------------------
480  // free workspace and return result
481  //--------------------------------------------------------------------------
482 
483  LG_FREE_ALL ;
484  return (GrB_SUCCESS) ;
