14 // The version number and date can also be obtained via compile-time constants
15 // from LAGraph.h. However, it is possible to compile a user application that
16 // #includes one version of LAGraph.h and then links with another version of
17 // the LAGraph library later on, so the version number and date may differ from
18 // the compile-time constants.
19 
20 // The LAGraph_Version method allows the library itself to be
21 // queried, after it is linked in with the user application.
22 
23 // The version_number array is set to LAGRAPH_VERSION_MAJOR,
24 // LAGRAPH_VERSION_MINOR, and LAGRAPH_VERSION_UPDATE, in that order.
25 // The LAGRAPH_DATE string is copied into the user-provided version_date
26 // string, and is null-terminated.
27 
28 #include "LG_internal.h"
29 
30 int LAGraph_Version
31 (
32  // output:
33  int version_number [3], // user-provided array of size 3
34  char version_date [LAGRAPH_MSG_LEN], // user-provided array
35  char *msg
36 )
37 {
38 
39  LG_CLEAR_MSG ;
40 
41  // check inputs
42  LG_ASSERT (version_number != NULL && version_date != NULL,
43  GrB_NULL_POINTER) ;
44 
45  // get version number and date
46  version_number [0] = LAGRAPH_VERSION_MAJOR ;
47  version_number [1] = LAGRAPH_VERSION_MINOR ;
48  version_number [2] = LAGRAPH_VERSION_UPDATE ;
49  strncpy (version_date, LAGRAPH_DATE, LAGRAPH_MSG_LEN) ;
50 
51  // make sure the date is null-terminated
52  version_date [LAGRAPH_MSG_LEN-1] = '\0' ;
53 
54  return (GrB_SUCCESS) ;
55 }
