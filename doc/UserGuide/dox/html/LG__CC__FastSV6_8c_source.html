15 // This is an Advanced algorithm (G->is_symmetric_structure must be known),
16 // but it is not user-callable (see LAGr_ConnectedComponents instead).
17 
18 // Code is based on the algorithm described in the following paper:
19 // Zhang, Azad, Hu. FastSV: A Distributed-Memory Connected Component
20 // Algorithm with Fast Convergence (SIAM PP20)
21 
22 // A subsequent update to the algorithm is here (which might not be reflected
23 // in this code):
24 // Yongzhe Zhang, Ariful Azad, Aydin Buluc: Parallel algorithms for finding
25 // connected components using linear algebra. J. Parallel Distributed Comput.
26 // 144: 14-27 (2020).
27 
28 // Modified by Tim Davis, Texas A&M University: revised Reduce_assign to use
29 // purely GrB* and GxB* methods and the matrix C. Added warmup phase. Changed
30 // to use GxB pack/unpack instead of GxB import/export. Converted to use the
31 // LAGraph_Graph object. Exploiting iso status for the temporary matrices
32 // C and T.
33 
34 // The input graph G must be undirected, or directed and with an adjacency
35 // matrix that has a symmetric structure. Self-edges (diagonal entries) are
36 // OK, and are ignored. The values and type of A are ignored; just its
37 // structure is accessed.
38 
39 // NOTE: This function must not be called by multiple user threads at the same
40 // time on the same graph G, since it unpacks G->A and then packs it back when
41 // done. G->A is unchanged when the function returns, but during execution
42 // G->A is empty. This will be fixed once the todos are finished below, and
43 // G->A will then become a truly read-only object (assuming GrB_wait (G->A)
44 // has been done first).

55 static inline GrB_Info fastsv
56 (
57  GrB_Matrix A, // adjacency matrix, G->A or a subset of G->A
58  GrB_Vector parent, // parent vector
59  GrB_Vector mngp, // min neighbor grandparent
60  GrB_Vector *gp, // grandparent
61  GrB_Vector *gp_new, // new grandparent (swapped with gp)
62  GrB_Vector t, // workspace
63  GrB_BinaryOp eq, // GrB_EQ_(integer type)
64  GrB_BinaryOp min, // GrB_MIN_(integer type)
65  GrB_Semiring min_2nd, // GrB_MIN_SECOND_(integer type)
66  GrB_Matrix C, // C(i,j) present if i = Px (j)
67  GrB_Index **Cp, // 0:n, size n+1
68  GrB_Index **Px, // Px: non-opaque copy of parent vector, size n
69  void **Cx, // size 1, contents not accessed
70  char *msg
71 )
72 {
73  GrB_Index n ;
74  GRB_TRY (GrB_Vector_size (&n, parent)) ;
75  GrB_Index Cp_size = (n+1) * sizeof (GrB_Index) ;
76  GrB_Index Ci_size = n * sizeof (GrB_Index) ;
77  GrB_Index Cx_size = sizeof (bool) ;
78  bool iso = true, jumbled = false, done = false ;
79 
80  while (true)
81  {
82 
83  //----------------------------------------------------------------------
84  // hooking & shortcutting
85  //----------------------------------------------------------------------
86 
87  // mngp = min (mngp, A*gp) using the MIN_SECOND semiring
88  GRB_TRY (GrB_mxv (mngp, NULL, min, min_2nd, A, *gp, NULL)) ;
89 
90  //----------------------------------------------------------------------
91  // parent = min (parent, C*mngp) where C(i,j) is present if i=Px(j)
92  //----------------------------------------------------------------------
93 
94  // Reduce_assign: The Px array of size n is the non-opaque copy of the
95  // parent vector, where i = Px [j] if the parent of node j is node i.
96  // It can thus have duplicates. The vectors parent and mngp are full
97  // (all entries present). This function computes the following, which
98  // is done explicitly in the Reduce_assign function in LG_CC_Boruvka:
99  //
100  // for (j = 0 ; j < n ; j++)
101  // {
102  // uint64_t i = Px [j] ;
103  // parent [i] = min (parent [i], mngp [j]) ;
104  // }
105  //
106  // If C(i,j) is present where i == Px [j], then this can be written as:
107  //
108  // parent = min (parent, C*mngp)
109  //
110  // when using the min_2nd semiring. This can be done efficiently
111  // because C can be constructed in O(1) time and O(1) additional space
112  // (not counting the prior Cp, Px, and Cx arrays), when using the
113  // SuiteSparse pack/unpack move constructors. The min_2nd semiring
114  // ignores the values of C and operates only on the structure, so its
115  // values are not relevant. Cx is thus chosen as a GrB_BOOL array of
116  // size 1 where Cx [0] = false, so the all entries present in C are
117  // equal to false.
118 
119  // pack Cp, Px, and Cx into a matrix C with C(i,j) present if Px(j) == i
120  GRB_TRY (GxB_Matrix_pack_CSC (C, Cp, /* Px is Ci: */ Px, Cx,
121  Cp_size, Ci_size, Cx_size, iso, jumbled, NULL)) ;
122 
123  // parent = min (parent, C*mngp) using the MIN_SECOND semiring
124  GRB_TRY (GrB_mxv (parent, NULL, min, min_2nd, C, mngp, NULL)) ;
125 
126  // unpack the contents of C, to make Px available to this method again.
127  GRB_TRY (GxB_Matrix_unpack_CSC (C, Cp, Px, Cx,
128  &Cp_size, &Ci_size, &Cx_size, &iso, &jumbled, NULL)) ;
129 
130  //----------------------------------------------------------------------
131  // parent = min (parent, mngp, gp)
132  //----------------------------------------------------------------------
133 
134  GRB_TRY (GrB_eWiseAdd (parent, NULL, min, min, mngp, *gp, NULL)) ;
135 
136  //----------------------------------------------------------------------
137  // calculate grandparent: gp_new = parent (parent), and extract Px
138  //----------------------------------------------------------------------
139 
140  // if parent is uint32, GraphBLAS typecasts to uint64 for Px.
141  GRB_TRY (GrB_Vector_extractTuples (NULL, *Px, &n, parent)) ;
142  GRB_TRY (GrB_extract (*gp_new, NULL, NULL, parent, *Px, n, NULL)) ;
143 
144  //----------------------------------------------------------------------
145  // terminate if gp and gp_new are the same
146  //----------------------------------------------------------------------
147 
148  GRB_TRY (GrB_eWiseMult (t, NULL, NULL, eq, *gp_new, *gp, NULL)) ;
149  GRB_TRY (GrB_reduce (&done, NULL, GrB_LAND_MONOID_BOOL, t, NULL)) ;
150  if (done) break ;
151 
152  // swap gp and gp_new
153  GrB_Vector s = (*gp) ; (*gp) = (*gp_new) ; (*gp_new) = s ;
154  }
155  return (GrB_SUCCESS) ;
156 }

198 int LG_CC_FastSV6 // SuiteSparse:GraphBLAS method, with GxB extensions
199 (
200  // output:
201  GrB_Vector *component, // component(i)=r if node is in the component r
202  // input:
203  LAGraph_Graph G, // input graph (modified then restored)
204  char *msg
205 )
206 {
207 
208 #if !LAGRAPH_SUITESPARSE
209  LG_ASSERT (false, GrB_NOT_IMPLEMENTED) ;
210 #else
211 
212  //--------------------------------------------------------------------------
213  // check inputs
214  //--------------------------------------------------------------------------
215 
216  LG_CLEAR_MSG ;
217 
218  int64_t *range = NULL ;
219  GrB_Index n, nvals, Cp_size = 0, *ht_key = NULL, *Px = NULL, *Cp = NULL,
220  *count = NULL, *Tp = NULL, *Tj = NULL ;
221  GrB_Vector parent = NULL, gp_new = NULL, mngp = NULL, gp = NULL, t = NULL,
222  y = NULL ;
223  GrB_Matrix T = NULL, C = NULL ;
224  void *Tx = NULL, *Cx = NULL ;
225  int *ht_count = NULL ;
226 
227  LG_TRY (LAGraph_CheckGraph (G, msg)) ;
228  LG_ASSERT (component != NULL, GrB_NULL_POINTER) ;
229 
230  LG_ASSERT_MSG ((G->kind == LAGraph_ADJACENCY_UNDIRECTED ||
231  (G->kind == LAGraph_ADJACENCY_DIRECTED &&
232  G->is_symmetric_structure == LAGraph_TRUE)),
233  LAGRAPH_SYMMETRIC_STRUCTURE_REQUIRED,
234  "G->A must be known to be symmetric") ;
235 
236  //--------------------------------------------------------------------------
237  // initializations
238  //--------------------------------------------------------------------------
239 
240  GrB_Matrix A = G->A ;
241  GRB_TRY (GrB_Matrix_nrows (&n, A)) ;
242  GRB_TRY (GrB_Matrix_nvals (&nvals, A)) ;
243 
244  // determine the integer type, operators, and semirings to use
245  GrB_Type Uint, Int ;
246  GrB_IndexUnaryOp ramp ;
247  GrB_Semiring min_2nd, min_2ndi ;
248  GrB_BinaryOp min, eq, imin ;
249  #ifdef COVERAGE
250  // Just for test coverage, use 64-bit ints for n > 100. Do not use this
251  // rule in production!
252  #define NBIG 100
253  #else
254  // For production use: 64-bit integers if n > 2^31
255  #define NBIG INT32_MAX
256  #endif
257  if (n > NBIG)
258  {
259  // use 64-bit integers throughout
260  Uint = GrB_UINT64 ;
261  Int = GrB_INT64 ;
262  ramp = GrB_ROWINDEX_INT64 ;
263  min = GrB_MIN_UINT64 ;
264  imin = GrB_MIN_INT64 ;
265  eq = GrB_EQ_UINT64 ;
266  min_2nd = GrB_MIN_SECOND_SEMIRING_UINT64 ;
267  min_2ndi = GxB_MIN_SECONDI_INT64 ;
268  }
269  else
270  {
271  // use 32-bit integers, except for Px and for constructing the matrix C
272  Uint = GrB_UINT32 ;
273  Int = GrB_INT32 ;
274  ramp = GrB_ROWINDEX_INT32 ;
275  min = GrB_MIN_UINT32 ;
276  imin = GrB_MIN_INT32 ;
277  eq = GrB_EQ_UINT32 ;
278  min_2nd = GrB_MIN_SECOND_SEMIRING_UINT32 ;
279  min_2ndi = GxB_MIN_SECONDI_INT32 ;
280  }
281 
282  // FASTSV_SAMPLES: number of samples to take from each row A(i,:).
283  // Sampling is used if the average degree is > 8 and if n > 1024.
284  #define FASTSV_SAMPLES 4
285  bool sampling = (nvals > n * FASTSV_SAMPLES * 2 && n > 1024) ;
286 
287 // [ todo: nthreads will not be needed once GxB_select with a GxB_RankUnaryOp
288 // and a new GxB_extract are added to SuiteSparse:GraphBLAS.
289  // determine # of threads to use
290  int nthreads, nthreads_outer, nthreads_inner ;
291  LG_TRY (LAGraph_GetNumThreads (&nthreads_outer, &nthreads_inner, msg)) ;
292  nthreads = nthreads_outer * nthreads_inner ;
293  nthreads = LAGRAPH_MIN (nthreads, n / 16) ;
294  nthreads = LAGRAPH_MAX (nthreads, 1) ;
295 // ]
296 
297  LG_TRY (LAGraph_Calloc ((void **) &Cx, 1, sizeof (bool), msg)) ;
298  LG_TRY (LAGraph_Malloc ((void **) &Px, n, sizeof (GrB_Index), msg)) ;
299 
300  // create Cp = 0:n (always 64-bit) and the empty C matrix
301  GRB_TRY (GrB_Matrix_new (&C, GrB_BOOL, n, n)) ;
302  GRB_TRY (GrB_Vector_new (&t, GrB_INT64, n+1)) ;
303  GRB_TRY (GrB_assign (t, NULL, NULL, 0, GrB_ALL, n+1, NULL)) ;
304  GRB_TRY (GrB_apply (t, NULL, NULL, GrB_ROWINDEX_INT64, t, 0, NULL)) ;
305  GRB_TRY (GxB_Vector_unpack_Full (t, (void **) &Cp, &Cp_size, NULL, NULL)) ;
306  GRB_TRY (GrB_free (&t)) ;
307 
308  //--------------------------------------------------------------------------
309  // warmup: parent = min (0:n-1, A*1) using the MIN_SECONDI semiring
310  //--------------------------------------------------------------------------
311 
312  // y (i) = min (i, j) for all entries A(i,j). This warmup phase takes only
313  // O(n) time, because of how the MIN_SECONDI semiring is implemented in
314  // SuiteSparse:GraphBLAS. A is held by row, and the first entry in A(i,:)
315  // is the minimum index j, so only the first entry in A(i,:) needs to be
316  // considered for each row i.
317 
318  GRB_TRY (GrB_Vector_new (&t, Int, n)) ;
319  GRB_TRY (GrB_Vector_new (&y, Int, n)) ;
320  GRB_TRY (GrB_assign (t, NULL, NULL, 0, GrB_ALL, n, NULL)) ;
321  GRB_TRY (GrB_assign (y, NULL, NULL, 0, GrB_ALL, n, NULL)) ;
322  GRB_TRY (GrB_apply (y, NULL, NULL, ramp, y, 0, NULL)) ;
323  GRB_TRY (GrB_mxv (y, NULL, imin, min_2ndi, A, t, NULL)) ;
324  GRB_TRY (GrB_free (&t)) ;
325 
326  // The typecast from Int to Uint is required because the ROWINDEX operator
327  // and MIN_SECONDI do not work in the UINT* domains, as built-in operators.
328  // parent = (Uint) y
329  GRB_TRY (GrB_Vector_new (&parent, Uint, n)) ;
330  GRB_TRY (GrB_assign (parent, NULL, NULL, y, GrB_ALL, n, NULL)) ;
331  GRB_TRY (GrB_free (&y)) ;
332 
333  // copy parent into gp, mngp, and Px. Px is a non-opaque 64-bit copy of the
334  // parent GrB_Vector. The Px array is always of type GrB_Index since it
335  // must be used as the input array for extractTuples and as Ci for pack_CSR.
336  // If parent is uint32, GraphBLAS typecasts it to the uint64 Px array.
337  GRB_TRY (GrB_Vector_extractTuples (NULL, Px, &n, parent)) ;
338  GRB_TRY (GrB_Vector_dup (&gp, parent)) ;
339  GRB_TRY (GrB_Vector_dup (&mngp, parent)) ;
340  GRB_TRY (GrB_Vector_new (&gp_new, Uint, n)) ;
341  GRB_TRY (GrB_Vector_new (&t, GrB_BOOL, n)) ;
342 
343  //--------------------------------------------------------------------------
344  // sample phase
345  //--------------------------------------------------------------------------
346 
347  if (sampling)
348  {
349 
350 // [ todo: GxB_select, using a new operator: GxB_RankUnaryOp, will do all this,
351 // with GxB_Matrix_select_RankOp_Scalar with operator GxB_LEASTRANK and a
352 // GrB_Scalar input equal to FASTSV_SAMPLES. Built-in operators will be,
353 // (where y is INT64):
354 //
355 // GxB_LEASTRANK (aij, i, j, k, d, y): select if aij has rank k <= y
356 // GxB_NLEASTRANK: select if aij has rank k > y
357 // GxB_GREATESTRANK (...) select if aij has rank k >= (d-y) where
358 // d = # of entries in A(i,:).
359 // GxB_NGREATESTRANK (...): select if aij has rank k < (d-y)
360 // and perhaps other operators such as:
361 // GxB_LEASTRELRANK (...): select aij if rank k <= y*d where y is double
362 // GxB_GREATESTRELRANK (...): select aij rank k > y*d where y is double
363 //
364 // By default, the rank of aij is its relative position as the kth entry in its
365 // row (from "left" to "right"). If a new GxB setting in the descriptor is
366 // set, then k is the relative position of aij as the kth entry in its column.
367 // The default would be that the rank is the position of aij in its row A(i,:).
368 
369 // Other:
370 // give me 3 random items from the row (y = 3)
371 // give me the 4 biggest *values* in each row (y = 4)
372 
373 // mxv:
374 // C = A*diag(D)
375 
376  //----------------------------------------------------------------------
377  // unpack A in CSR format
378  //----------------------------------------------------------------------
379 
380  void *Ax ;
381  GrB_Index *Ap, *Aj, Ap_size, Aj_size, Ax_size ;
382  bool A_jumbled, A_iso ;
383  GRB_TRY (GxB_Matrix_unpack_CSR (A, &Ap, &Aj, &Ax,
384  &Ap_size, &Aj_size, &Ax_size, &A_iso, &A_jumbled, NULL)) ;
385 
386  //----------------------------------------------------------------------
387  // allocate workspace, including space to construct T
388  //----------------------------------------------------------------------
389 
390  GrB_Index Tp_size = (n+1) * sizeof (GrB_Index) ;
391  GrB_Index Tj_size = nvals * sizeof (GrB_Index) ;
392  GrB_Index Tx_size = sizeof (bool) ;
393  LG_TRY (LAGraph_Malloc ((void **) &Tp, n+1, sizeof (GrB_Index), msg)) ;
394  LG_TRY (LAGraph_Malloc ((void **) &Tj, nvals, sizeof (GrB_Index),
395  msg)) ;
396  LG_TRY (LAGraph_Calloc ((void **) &Tx, 1, sizeof (bool), msg)) ;
397  LG_TRY (LAGraph_Malloc ((void **) &range, nthreads + 1,
398  sizeof (int64_t), msg)) ;
399  LG_TRY (LAGraph_Calloc ((void **) &count, nthreads + 1,
400  sizeof (GrB_Index), msg)) ;
401 
402  //----------------------------------------------------------------------
403  // define parallel tasks to construct T
404  //----------------------------------------------------------------------
405 
406  // thread tid works on rows range[tid]:range[tid+1]-1 of A and T
407  for (int tid = 0 ; tid <= nthreads ; tid++)
408  {
409  range [tid] = (n * tid + nthreads - 1) / nthreads ;
410  }
411 
412  //----------------------------------------------------------------------
413  // determine the number entries to be constructed in T for each thread
414  //----------------------------------------------------------------------
415 
416  #pragma omp parallel for num_threads(nthreads) schedule(static)
417  for (int tid = 0 ; tid < nthreads ; tid++)
418  {
419  for (int64_t i = range [tid] ; i < range [tid+1] ; i++)
420  {
421  int64_t deg = Ap [i + 1] - Ap [i] ;
422  count [tid + 1] += LAGRAPH_MIN (FASTSV_SAMPLES, deg) ;
423  }
424  }
425 
426  //----------------------------------------------------------------------
427  // count = cumsum (count)
428  //----------------------------------------------------------------------
429 
430  for (int tid = 0 ; tid < nthreads ; tid++)
431  {
432  count [tid + 1] += count [tid] ;
433  }
434 
435  //----------------------------------------------------------------------
436  // construct T
437  //----------------------------------------------------------------------
438 
439  // T (i,:) consists of the first FASTSV_SAMPLES of A (i,:).
440 
441  #pragma omp parallel for num_threads(nthreads) schedule(static)
442  for (int tid = 0 ; tid < nthreads ; tid++)
443  {
444  GrB_Index p = count [tid] ;
445  Tp [range [tid]] = p ;
446  for (int64_t i = range [tid] ; i < range [tid+1] ; i++)
447  {
448  // construct T (i,:) from the first entries in A (i,:)
449  for (int64_t j = 0 ;
450  j < FASTSV_SAMPLES && Ap [i] + j < Ap [i + 1] ; j++)
451  {
452  Tj [p++] = Aj [Ap [i] + j] ;
453  }
454  Tp [i + 1] = p ;
455  }
456  }
457 
458  //----------------------------------------------------------------------
459  // import the result into the GrB_Matrix T
460  //----------------------------------------------------------------------
461 
462  GRB_TRY (GrB_Matrix_new (&T, GrB_BOOL, n, n)) ;
463  GRB_TRY (GxB_Matrix_pack_CSR (T, &Tp, &Tj, &Tx, Tp_size, Tj_size,
464  Tx_size, /* T is iso: */ true, A_jumbled, NULL)) ;
465 
466 // ] todo: the above will all be done as a single call to GxB_select.
467 
468  //----------------------------------------------------------------------
469  // find the connected components of T
470  //----------------------------------------------------------------------
471 
472  GRB_TRY (fastsv (T, parent, mngp, &gp, &gp_new, t, eq, min, min_2nd,
473  C, &Cp, &Px, &Cx, msg)) ;
474 
475  //----------------------------------------------------------------------
476  // use sampling to estimate the largest connected component in T
477  //----------------------------------------------------------------------
478 
479  // The sampling below computes an estimate of the mode of the parent
480  // vector, the contents of which are currently in the non-opaque Px
481  // array.
482 
483  // hash table size must be a power of 2
484  #define HASH_SIZE 1024
485  // number of samples to insert into the hash table
486  #define HASH_SAMPLES 864
487  #define HASH(x) (((x << 4) + x) & (HASH_SIZE-1))
488  #define NEXT(x) ((x + 23) & (HASH_SIZE-1))
489 
490  // allocate and initialize the hash table
491  LG_TRY (LAGraph_Malloc ((void **) &ht_key, HASH_SIZE,
492  sizeof (GrB_Index), msg)) ;
493  LG_TRY (LAGraph_Calloc ((void **) &ht_count, HASH_SIZE,
494  sizeof (int), msg)) ;
495  for (int k = 0 ; k < HASH_SIZE ; k++)
496  {
497  ht_key [k] = UINT64_MAX ;
498  }
499 
500  // hash the samples and find the most frequent entry
501  uint64_t seed = n ; // random number seed
502  int64_t key = -1 ; // most frequent entry
503  int max_count = 0 ; // frequency of most frequent entry
504  for (int64_t k = 0 ; k < HASH_SAMPLES ; k++)
505  {
506  // select an entry from Px at random
507  GrB_Index x = Px [LG_Random60 (&seed) % n] ;
508  // find x in the hash table
509  GrB_Index h = HASH (x) ;
510  while (ht_key [h] != UINT64_MAX && ht_key [h] != x) h = NEXT (h) ;
511  // add x to the hash table
512  ht_key [h] = x ;
513  ht_count [h]++ ;
514  // keep track of the most frequent value
515  if (ht_count [h] > max_count)
516  {
517  key = ht_key [h] ;
518  max_count = ht_count [h] ;
519  }
520  }
521 
522  //----------------------------------------------------------------------
523  // compact the largest connected component in A
524  //----------------------------------------------------------------------
525 
526  // Construct a new matrix T from the input matrix A (the matrix A is
527  // not changed). The key node is the representative of the (estimated)
528  // largest component. T is constructed as a copy of A, except:
529  // (1) all edges A(i,:) for nodes i in the key component deleted, and
530  // (2) for nodes i not in the key component, A(i,j) is deleted if
531  // j is in the key component.
532  // (3) If A(i,:) has any deletions from (2), T(i,key) is added to T.
533 
534 // [ todo: replace this with GxB_extract with GrB_Vector index arrays.
535 // See https://github.com/GraphBLAS/graphblas-api-c/issues/67 .
536 // This method will not insert the new entries T(i,key) for rows i that have
537 // had entries deleted. That can be done with GrB_assign, with an n-by-1 mask
538 // M computed from the before-and-after row degrees of A and T:
539 // M = (parent != key) && (out_degree(T) < out_degree(A))
540 // J [0] = key.
541 // GxB_Matrix_subassign_BOOL (T, M, NULL, true, GrB_ALL, n, J, 1, NULL)
542 // or with
543 // GrB_Col_assign (T, M, NULL, t, GrB_ALL, j, NULL) with an all-true
544 // vector t.
545
