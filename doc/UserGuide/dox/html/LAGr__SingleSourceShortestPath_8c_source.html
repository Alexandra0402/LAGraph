27 // LAGr_SingleSourceShortestPath computes the shortest path lengths from the
28 // specified source vertex to all other vertices in the graph.
29 
30 // The parent vector is not computed; see LAGraph_BF_* instead.
31 
32 // NOTE: this method gets stuck in an infinite loop when there are negative-
33 // weight cycles in the graph.

59 #define setelement(s, k) \
60 { \
61  switch (tcode) \
62  { \
63  default: \
64  case 0 : GrB_Scalar_setElement_INT32 (s, k * delta_int32 ) ; break ; \
65  case 1 : GrB_Scalar_setElement_INT64 (s, k * delta_int64 ) ; break ; \
66  case 2 : GrB_Scalar_setElement_UINT32 (s, k * delta_uint32) ; break ; \
67  case 3 : GrB_Scalar_setElement_UINT64 (s, k * delta_uint64) ; break ; \
68  case 4 : GrB_Scalar_setElement_FP32 (s, k * delta_fp32 ) ; break ; \

73 int LAGr_SingleSourceShortestPath
74 (
75  // output:
76  GrB_Vector *path_length, // path_length (i) is the length of the shortest
77  // path from the source vertex to vertex i
78  // input:
79  const LAGraph_Graph G, // input graph, not modified
80  GrB_Index source, // source vertex
81  GrB_Scalar Delta, // delta value for delta stepping
82  char *msg
83 )
84 {
85 
86  //--------------------------------------------------------------------------
87  // check inputs
88  //--------------------------------------------------------------------------
89 
90  LG_CLEAR_MSG ;
91  GrB_Scalar lBound = NULL ; // the threshold for GrB_select
92  GrB_Scalar uBound = NULL ; // the threshold for GrB_select
93  GrB_Matrix AL = NULL ; // graph containing the light weight edges
94  GrB_Matrix AH = NULL ; // graph containing the heavy weight edges
95  GrB_Vector t = NULL ; // tentative shortest path length
96  GrB_Vector tmasked = NULL ;
97  GrB_Vector tReq = NULL ;
98  GrB_Vector tless = NULL ;
99  GrB_Vector s = NULL ;
100  GrB_Vector reach = NULL ;
101  GrB_Vector Empty = NULL ;
102 
103  LG_TRY (LAGraph_CheckGraph (G, msg)) ;
104  LG_ASSERT (path_length != NULL && Delta != NULL, GrB_NULL_POINTER) ;
105  (*path_length) = NULL ;
106 
107  GrB_Index nvals ;
108  LG_TRY (GrB_Scalar_nvals (&nvals, Delta)) ;
109  LG_ASSERT_MSG (nvals == 1, GrB_EMPTY_OBJECT, "Delta is missing") ;
110 
111  GrB_Matrix A = G->A ;
112  GrB_Index n ;
113  GRB_TRY (GrB_Matrix_nrows (&n, A)) ;
114  LG_ASSERT_MSG (source < n, GrB_INVALID_INDEX, "invalid source node") ;
115 
116  //--------------------------------------------------------------------------
117  // initializations
118  //--------------------------------------------------------------------------
119 
120  // get the type of the A matrix
121  GrB_Type etype ;
122  char typename [LAGRAPH_MAX_NAME_LEN] ;
123  LG_TRY (LAGraph_Matrix_TypeName (typename, A, msg)) ;
124  LG_TRY (LAGraph_TypeFromName (&etype, typename, msg)) ;
125 
126  GRB_TRY (GrB_Scalar_new (&lBound, etype)) ;
127  GRB_TRY (GrB_Scalar_new (&uBound, etype)) ;
128  GRB_TRY (GrB_Vector_new (&t, etype, n)) ;
129  GRB_TRY (GrB_Vector_new (&tmasked, etype, n)) ;
130  GRB_TRY (GrB_Vector_new (&tReq, etype, n)) ;
131  GRB_TRY (GrB_Vector_new (&Empty, GrB_BOOL, n)) ;
132  GRB_TRY (GrB_Vector_new (&tless, GrB_BOOL, n)) ;
133  GRB_TRY (GrB_Vector_new (&s, GrB_BOOL, n)) ;
134  GRB_TRY (GrB_Vector_new (&reach, GrB_BOOL, n)) ;
135 
136 #if LAGRAPH_SUITESPARSE
137  // optional hints for SuiteSparse:GraphBLAS
138  GRB_TRY (GxB_set (t, GxB_SPARSITY_CONTROL, GxB_BITMAP)) ;
139  GRB_TRY (GxB_set (tmasked, GxB_SPARSITY_CONTROL, GxB_SPARSE)) ;
140  GRB_TRY (GxB_set (tReq, GxB_SPARSITY_CONTROL, GxB_SPARSE)) ;
141  GRB_TRY (GxB_set (tless, GxB_SPARSITY_CONTROL, GxB_SPARSE)) ;
142  GRB_TRY (GxB_set (s, GxB_SPARSITY_CONTROL, GxB_SPARSE)) ;
143  GRB_TRY (GxB_set (reach, GxB_SPARSITY_CONTROL, GxB_BITMAP)) ;
144 #endif
145 
146  // select the operators, and set t (:) = infinity
147  GrB_IndexUnaryOp ne, le, ge, lt, gt ;
148  GrB_BinaryOp less_than ;
149  GrB_Semiring min_plus ;
150  int tcode ;
151  int32_t delta_int32 ;
152  int64_t delta_int64 ;
153  uint32_t delta_uint32 ;
154  uint64_t delta_uint64 ;
155  float delta_fp32 ;
156  double delta_fp64 ;
157 
158  bool negative_edge_weights = true ;
159 
160  if (etype == GrB_INT32)
161  {
162  GRB_TRY (GrB_Scalar_extractElement (&delta_int32, Delta)) ;
163  GRB_TRY (GrB_assign (t, NULL, NULL, (int32_t) INT32_MAX,
164  GrB_ALL, n, NULL)) ;
165  ne = GrB_VALUENE_INT32 ;
166  le = GrB_VALUELE_INT32 ;
167  ge = GrB_VALUEGE_INT32 ;
168  lt = GrB_VALUELT_INT32 ;
169  gt = GrB_VALUEGT_INT32 ;
170  less_than = GrB_LT_INT32 ;
171  min_plus = GrB_MIN_PLUS_SEMIRING_INT32 ;
172  tcode = 0 ;
173  }
174  else if (etype == GrB_INT64)
175  {
176  GRB_TRY (GrB_Scalar_extractElement (&delta_int64, Delta)) ;
177  GRB_TRY (GrB_assign (t, NULL, NULL, (int64_t) INT64_MAX,
178  GrB_ALL, n, NULL)) ;
179  ne = GrB_VALUENE_INT64 ;
180  le = GrB_VALUELE_INT64 ;
181  ge = GrB_VALUEGE_INT64 ;
182  lt = GrB_VALUELT_INT64 ;
183  gt = GrB_VALUEGT_INT64 ;
184  less_than = GrB_LT_INT64 ;
185  min_plus = GrB_MIN_PLUS_SEMIRING_INT64 ;
186  tcode = 1 ;
187  }
188  else if (etype == GrB_UINT32)
189  {
190  GRB_TRY (GrB_Scalar_extractElement (&delta_uint32, Delta)) ;
191  GRB_TRY (GrB_assign (t, NULL, NULL, (uint32_t) UINT32_MAX,
192  GrB_ALL, n, NULL)) ;
193  ne = GrB_VALUENE_UINT32 ;
194  le = GrB_VALUELE_UINT32 ;
195  ge = GrB_VALUEGE_UINT32 ;
196  lt = GrB_VALUELT_UINT32 ;
197  gt = GrB_VALUEGT_UINT32 ;
198  less_than = GrB_LT_UINT32 ;
199  min_plus = GrB_MIN_PLUS_SEMIRING_UINT32 ;
200  tcode = 2 ;
201  negative_edge_weights = false ;
202  }
203  else if (etype == GrB_UINT64)
204  {
205  GRB_TRY (GrB_Scalar_extractElement (&delta_uint64, Delta)) ;
206  GRB_TRY (GrB_assign (t, NULL, NULL, (uint64_t) UINT64_MAX,
207  GrB_ALL, n, NULL)) ;
208  ne = GrB_VALUENE_UINT64 ;
209  le = GrB_VALUELE_UINT64 ;
210  ge = GrB_VALUEGE_UINT64 ;
211  lt = GrB_VALUELT_UINT64 ;
212  gt = GrB_VALUEGT_UINT64 ;
213  less_than = GrB_LT_UINT64 ;
214  min_plus = GrB_MIN_PLUS_SEMIRING_UINT64 ;
215  tcode = 3 ;
216  negative_edge_weights = false ;
217  }
218  else if (etype == GrB_FP32)
219  {
220  GRB_TRY (GrB_Scalar_extractElement (&delta_fp32, Delta)) ;
221  GRB_TRY (GrB_assign (t, NULL, NULL, (float) INFINITY,
222  GrB_ALL, n, NULL)) ;
223  ne = GrB_VALUENE_FP32 ;
224  le = GrB_VALUELE_FP32 ;
225  ge = GrB_VALUEGE_FP32 ;
226  lt = GrB_VALUELT_FP32 ;
227  gt = GrB_VALUEGT_FP32 ;
228  less_than = GrB_LT_FP32 ;
229  min_plus = GrB_MIN_PLUS_SEMIRING_FP32 ;
230  tcode = 4 ;
231  }
232  else if (etype == GrB_FP64)
233  {
234  GRB_TRY (GrB_Scalar_extractElement (&delta_fp64, Delta)) ;
235  GRB_TRY (GrB_assign (t, NULL, NULL, (double) INFINITY,
236  GrB_ALL, n, NULL)) ;
237  ne = GrB_VALUENE_FP64 ;
238  le = GrB_VALUELE_FP64 ;
239  ge = GrB_VALUEGE_FP64 ;
240  lt = GrB_VALUELT_FP64 ;
241  gt = GrB_VALUEGT_FP64 ;
242  less_than = GrB_LT_FP64 ;
243  min_plus = GrB_MIN_PLUS_SEMIRING_FP64 ;
244  tcode = 5 ;
245  }
246  else
247  {
248  LG_ASSERT_MSG (false, GrB_NOT_IMPLEMENTED, "type not supported") ;
249  }
250 
251  // check if the graph might have negative edge weights
252  if (negative_edge_weights)
253  {
254  double emin = -1 ;
255  if (G->emin != NULL &&
256  (G->emin_state == LAGraph_VALUE ||
257  G->emin_state == LAGraph_BOUND))
258  {
259  GRB_TRY (GrB_Scalar_extractElement_FP64 (&emin, G->emin)) ;
260  }
261 // else
262 // {
263 // // a future Basic algorithm should compute G->emin and perhaps
264 // // G->emax, then compute Delta automatically.
265 // }
266  negative_edge_weights = (emin < 0) ;
267  }
268 
269  // t (src) = 0
270  GRB_TRY (GrB_Vector_setElement (t, 0, source)) ;
271 
272  // reach (src) = true
273  GRB_TRY (GrB_Vector_setElement (reach, true, source)) ;
274 
275  // s (src) = true
276  GRB_TRY (GrB_Vector_setElement (s, true, source)) ;
277 
278  // AL = A .* (A <= Delta)
279  GRB_TRY (GrB_Matrix_new (&AL, etype, n, n)) ;
280  GRB_TRY (GrB_select (AL, NULL, NULL, le, A, Delta, NULL)) ;
281  GRB_TRY (GrB_wait (AL, GrB_MATERIALIZE)) ;
282 
283  // FUTURE: costly for some problems, taking up to 50% of the total time:
284  // AH = A .* (A > Delta)
285  GRB_TRY (GrB_Matrix_new (&AH, etype, n, n)) ;
286  GRB_TRY (GrB_select (AH, NULL, NULL, gt, A, Delta, NULL)) ;
287  GRB_TRY (GrB_wait (AH, GrB_MATERIALIZE)) ;
288 
289  //--------------------------------------------------------------------------
290  // while (t >= step*Delta) not empty
291  //--------------------------------------------------------------------------
292 
293  for (int64_t step = 0 ; ; step++)
294  {
295 
296  //----------------------------------------------------------------------
297  // tmasked = all entries in t<reach> that are less than (step+1)*Delta
298  //----------------------------------------------------------------------
299 
300  setelement (uBound, (step+1)) ; // uBound = (step+1) * Delta
301  GRB_TRY (GrB_Vector_clear (tmasked)) ;
302 
303  // tmasked<reach> = t
304  // FUTURE: this is costly, typically using Method 06s in SuiteSparse,
305  // which is a very general-purpose one. Write a specialized kernel to
306  // exploit the fact that reach and t are bitmap and tmasked starts
307  // empty, or fuse this assignment with the GrB_select below.
308  GRB_TRY (GrB_assign (tmasked, reach, NULL, t, GrB_ALL, n, NULL)) ;
309  // tmasked = select (tmasked < (step+1)*Delta)
310  GRB_TRY (GrB_select (tmasked, NULL, NULL, lt, tmasked, uBound, NULL)) ;
311  // --- alternative:
312  // FUTURE this is slower than the above but should be much faster.
313  // GrB_select is computing a bitmap result then converting it to
314  // sparse. t and reach are both bitmap and tmasked finally sparse.
315  // tmasked<reach> = select (t < (step+1)*Delta)
316  // GRB_TRY (GrB_select (tmasked, reach, NULL, lt, t, uBound, NULL)) ;
317 
318  GrB_Index tmasked_nvals ;
319  GRB_TRY (GrB_Vector_nvals (&tmasked_nvals, tmasked)) ;
320 
321  //----------------------------------------------------------------------
322  // continue while the current bucket (tmasked) is not empty
323  //----------------------------------------------------------------------
324 
325  while (tmasked_nvals > 0)
326  {
327  // tReq = AL'*tmasked using the min_plus semiring
328  GRB_TRY (GrB_vxm (tReq, NULL, NULL, min_plus, tmasked, AL, NULL)) ;
329 
330  // s<struct(tmasked)> = true
331  GRB_TRY (GrB_assign (s, tmasked, NULL, (bool) true, GrB_ALL, n,
332  GrB_DESC_S)) ;
333 
334  // if nvals (tReq) is 0, no need to continue the rest of this loop
335  GrB_Index tReq_nvals ;
336  GRB_TRY (GrB_Vector_nvals (&tReq_nvals, tReq)) ;
337  if (tReq_nvals == 0) break ;
338 
339  // tless = (tReq .< t) using set intersection
340  GRB_TRY (GrB_eWiseMult (tless, NULL, NULL, less_than, tReq, t,
341  NULL)) ;
342 
343  // remove explicit zeros from tless so it can be used as a
344  // structural mask
345  GrB_Index tless_nvals ;
346  GRB_TRY (GrB_select (tless, NULL, NULL, ne, tless, 0, NULL)) ;
347  GRB_TRY (GrB_Vector_nvals (&tless_nvals, tless)) ;
348  if (tless_nvals == 0) break ;
349 
350  // update reachable node list/mask
351  // reach<struct(tless)> = true
352  GRB_TRY (GrB_assign (reach, tless, NULL, (bool) true, GrB_ALL, n,
353  GrB_DESC_S)) ;
354 
355  // tmasked<struct(tless)> = select (tReq < (step+1)*Delta)
356  GRB_TRY (GrB_Vector_clear (tmasked)) ;
357  GRB_TRY (GrB_select (tmasked, tless, NULL, lt, tReq, uBound,
358  GrB_DESC_S)) ;
359 
360  // For general graph with some negative weights:
361  if (negative_edge_weights)
362  {
363  // If all entries of the graph are known to be positive, and
364  // the entries of tmasked are at least step*Delta, tReq =
365  // tmasked min.+ AL must be >= step*Delta. Therefore, there is
366  // no need to perform this GrB_select with ge to find tmasked
367  // >= step*Delta from tReq.
368  setelement (lBound, (step)) ; // lBound = step*Delta
369  // tmasked = select entries in tmasked that are >= step*Delta
370  GRB_TRY (GrB_select (tmasked, NULL, NULL, ge, tmasked, lBound,
371  NULL)) ;
372  }
373 
374  // t<struct(tless)> = tReq
375  GRB_TRY (GrB_assign (t, tless, NULL, tReq, GrB_ALL, n, GrB_DESC_S));
376  GRB_TRY (GrB_Vector_nvals (&tmasked_nvals, tmasked)) ;
377  }
378 
379  // tmasked<s> = t
380  GRB_TRY (GrB_Vector_clear (tmasked)) ;
381  GRB_TRY (GrB_assign (tmasked, s, NULL, t, GrB_ALL, n, GrB_DESC_S)) ;
382 
383  // tReq = AH'*tmasked using the min_plus semiring
384  GRB_TRY (GrB_vxm (tReq, NULL, NULL, min_plus, tmasked, AH, NULL)) ;
385 
386  // tless = (tReq .< t) using set intersection
387  GRB_TRY (GrB_eWiseMult (tless, NULL, NULL, less_than, tReq, t, NULL)) ;
388 
389  // t<tless> = tReq, which computes t = min (t, tReq)
390  GRB_TRY (GrB_assign (t, tless, NULL, tReq, GrB_ALL, n, NULL)) ;
391 
392  //----------------------------------------------------------------------
393  // find out how many left to be computed
394  //----------------------------------------------------------------------
395 
396  // update reachable node list
397  // reach<tless> = true
398  GRB_TRY (GrB_assign (reach, tless, NULL, (bool) true, GrB_ALL, n,
399  NULL)) ;
400 
401  // remove previous buckets
402  // reach<struct(s)> = Empty
403  GRB_TRY (GrB_assign (reach, s, NULL, Empty, GrB_ALL, n, GrB_DESC_S)) ;
404  GrB_Index nreach ;
405  GRB_TRY (GrB_Vector_nvals (&nreach, reach)) ;
406  if (nreach == 0) break ;
407 
408  GRB_TRY (GrB_Vector_clear (s)) ; // clear s for the next iteration
409  }
410 
411  //--------------------------------------------------------------------------
412  // free workspace and return result
413  //--------------------------------------------------------------------------
414 
415  (*path_length) = t ;
416  LG_FREE_WORK ;
417  return (GrB_SUCCESS) ;
418 }
